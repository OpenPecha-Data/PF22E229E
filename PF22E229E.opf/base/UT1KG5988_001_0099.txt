༄༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྻ་དྲུ་མ་སཱུ་ཏྲ། བོད་སྐད་དུ། འཕགས་པ་ལྗོན་ཤིང་གི་མདོ། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །ཇི་ལྟར་དབྱར་གྱི་ལྗོན་ཤིང་བཟང་པོ་ལ། །འབྲས་བུ་ལོ་མ་མེ་ཏོག་འདབ་རྒྱས་ཀྱང༌། །གྲངས་ལྷགས་ཐན་པ་དུས་ཀྱི་རླུང་བྱུང་ཚེ། །སྐད་ཅིག་ཙམ་ལ་མེད་པར་འགྱུར་བ་བཞིན། །ཅིས་ཀྱང་མི་སྡོད་སྒྱུ་མའི་རང་བཞིན་ལ། །རང་བཞིན་ཐ་མལ་དེ་བཞིན་ཤེས་པར་བྱ། །དབྱུག་པས་རྒྱབ་ཀྱང་རང་ལ་འབད་པའི་སེམས་མེད་ན། །བྱོལ་སོང་ཕྱུགས་ཀྱི་བུ་དང་ཅིས་མ་ཁྱད། །བསམ་པ་མེད་པར་ཉལ་ཏེ་གཉིད་ལོག་ན། །འཇིག་རྟེན་འདི་ཡི་ཉི་མ་སྒོ་རུ་ནུབ། །སྔོན་གྱི་བསོད་ནམས་ཅི་བསགས་ཟད་དུ་ཉེ། །རང་གིས་རང་ལ་གཞེན་བསྐུལ་མ་བཏབ་ན། །དེ་ལ་གཞན་གྱིས་གཞེན་བསྐུལ་འདེབས་པ་སུ། །དགེ་བའི་ལས་རྣམས་ཕལ་ཆེར་སྤྱད་པ་ནི། །ཞིང་རབ་གཤིན་ལ་ས་བོན་བཏབ་པ་བཞིན། །དཔྱིད་ན་སྣང་བ་མེད་ཀྱང་སྟོན་ན་འབྲས་བུ་སྨིན། །སྡིག་པའི་ལས་རྣམས་ཕལ་ཆེར་སྤྱད་པ་ནི། །གང་ལྟར་འགྲོ་བའི་ས་ཕྱོགས་དེ་ན་སྡོད། །ཕ་མའི་དོན་དུ་སྡིག་པའི་ལས་མི་བྱ། །ལུས་དང་སྲོག་གི་ཕྱིར་ཡང་དམ་པའི་ཆོས་མི་སྤོང༌། །ད་ལྟ་ཆོས་སྤྱོད་ཁོམ་པའི་དུས་ན་བགྱི། །དགེ་བའི་ལས་རྣམས་ཕལ་ཆེར་སྤྱད་པ་ནི། །གང་ལྟར་འགྲོ་བའི་ས་ཕྱོགས་དེ་ན་སྡོད། །དགེ་བ་བཅུ་ཡི་མལ་ཚེ་དམ་དུ་སྡོམས། །བླ་མེད་བྱང་ཆུབ་སངས་རྒྱས་དང༌། །མཐོ་རིས་སྐྱེ་བ་ལྷ་དང་མི། །ཡི་དྭགས་བྱོལ་སོང་དམྱལ་བ་གསུམ། །གང་འདམ་རང་ལ་དབང་ཡོད་ན། །རང་གིས་རང་ལ་གནོད་མ་བྱེད། །དུག་དང་བདུད་རྩི་གང་ཕན་པ། །བལྟས་ན་གསལ་ཏེ་མི་མངོན་ནམ། །བླུན་པོ་བྱིས་པ་ལྟ་བུའི་བློ། །འབྲས་བུ་འདོད་པས་བསླུས་ན་ནི། །ངན་སོང་གསུམ་གྱི་གནས་ཁང་དུ། །སླེབ་ནས་འགྱོད་པས་ཅི་ལ་ཕན། །ཇི་ལྟར་གཉེན་འབྲེལ་མང་ལ་དབང་ཆེ་ཡང༌། །ཤི་ནས་གྲོགས་ལ་འགྲོ་རུ་མེད། །དེ་ཡི་དུས་ན་དཔའ་དར་མེད།

བདག་ཉིད་མལ་ན་ཉལ་བཞིན་དུ། །གཉེན་བཤེས་མང་པོས་མཐའ་བསྐོར་ཡང༌། །གཤིན་རྗེའི་ཕོ་ཉས་ཟིན་པའི་ཚེ། །འཆི་བ་གཅིག་པུས་མྱོང་བར་འགྱུར། །དེ་ཚེ་གཉེན་མང་ཅི་ལ་ཕན། །མཛངས་པ་ལྷ་ཆོས་མི་བྱེད་ན། །དེ་ནི་གྲུང་པོ་མདོ་མེད་ཡིན། །ཕྱུག་པོ་སྦྱིན་པ་མི་གཏོང་ན། །དེ་ནི་གཞན་གྱི་གཏེར་བསྲུང་ཡིན། །མཁས་པ་དམ་ཚིག་མི་བསྲུང་ན། །དེ་ནི་སྡོང་པོ་ཁོང་རུལ་ཡིན། །ཧ་ཅང་གསོག་ལ་མ་ཞེན་ཅིག །ཧ་ཅང་གསོག་ལ་ཞེན་ཆེས་ན། །བསགས་པ་གཞན་གྱིས་འཁྱེར་དུ་མཆི། །སྤྲང་བུས་སྦྲང་རྩི་བསགས་པ་དེ། །དེ་དག་གཞན་གྱིས་སྤྱོད་པར་བྱེད། །གར་སྐྱེ་སུ་ལའང་ཆ་མེད་ཀྱིས། །རང་གི་ལུས་ལ་དཔེ་ལོང་ལས། །གཞན་གྱི་ལུས་ལ་གནོད་མ་བྱེད། །འབད་པ་མེད་པའི་ཏིལ་དག་ལ། །ཏིལ་མར་འབྱུང་བ་ག་ལ་འགྱུར། །དཔེར་ན་གྲོ་སྲན་ས་བོན་ལས། །འབྲས་བུ་ནས་སུ་ག་ལ་འགྱུར། །ལེགས་པར་སྤྱད་ན་དཔལ་དུ་ཆེ། །ཉེས་པར་སྤྱད་ན་བྱུར་དུ་ཆེ། །སོ་སོ་སྐྱེ་བོ་བློ་ངན་པ། །མུ་སྟེགས་ལ་ནི་ཡིད་ཆེས་ཤིང༌། །ལས་ཀྱི་དོན་དུ་མཚན་མ་སྤྱོད། །མཚན་མས་གྲོལ་བར་ག་ལ་འགྱུར། །དཔེར་ན་སྦུར་འགྲོ་འདོད་པ་ལ། །བལ་མདུད་ནང་དུ་འཐུམས་པ་བཞིན། །སུག་དྲུག་བསྐྱོད་པས་ཕྱི་ཕྱིར་བསྡམས། །མོ་བོན་དག་ལ་སྲིད་མ་ལྟོས། །འདྲེ་སྲིན་དག་ལ་ཡས་མ་འབོད། །བདུད་དང་བགེགས་ལ་སྐྱབས་མ་འཚོལ། །སྐྱེ་ཤིའི་མགོན་དུ་སངས་རྒྱས་བཟང༌། །སྡུག་བསྔལ་སྐྱབས་སུ་དམ་ཆོས་བཟང༌། །འཇིགས་པའི་མགོན་དུ་དགེ་འདུན་ཆེ། །བློ་གཏད་དཀོན་མཆོག་གསུམ་ལ་གྱིས། །ཡང་དག་བདུད་རྩིའི་སྨན་རྣམས་འཐུང༌། །ལོག་པར་ལྟ་བའི་དུག་མ་འཐུངས། །བྱིས་པ་གཞོན་ནུའི་བསམ་པ་ལ། །གཞོན་ནུ་རྟག་པར་འདུག་སྙམ་སྟེ། །བྱིས་པ་གཞོན་ནུ་ཡོང་མི་རྟག །ཞག་ཟླ་ཟད་ཀྱིན་རྒས་ལ་ཉེ། །གསོད་སར་ཁྲིད་པའི་བཙོན་བཞིན་དུ། །གོམ་རེ་བོར་ཞིང་འཆི་ལ་ཉེ། །དེ་རིང་སང་འཆི་ཆ་མེད་ཀྱི། །ད་ལྟ་ཉིད་དུ་མྱུར་ཏེ་མཛོད། །དེང་འཆི་སང་འཆི་བདག་མི་ཚོར། །ཁྲི་འབོད་སྟོང་འབོད་རྣ་མི་ཐོས། །དགེ་བའི་བཤེས་དང་འབྲལ་དུ་ཉེ། །བདུད་ནི་གང་ལའང་གླགས་བལྟ་ཞིང༌། །གེགས་ནི་གང་ལའང་བར་དུ་གཅོད། །ཚེ་གང་བསགས་པའི་ཟས་དང་ནོར། །བུ་དང་ཆུང་མ་ཡུལ་འཁོར་རྣམས། །སློག་རུལ་བཞིན་དུ་བོར་ནས་ཀྱང༌། །གཅིག་པུར་མྱུར་དུ་འགྲོ་དགོས་ན། །བློ་གཏད་ཐུབ་པའི་གྲོགས་ཡོད་དམ། །ལོ་བརྒྱ་ཐུབ་པའི་ཟས་ཡོད་ཀྱང༌། །ཤི་བའི་ནངས་པར་ལྟོགས་པར་འགྲོ། །ལོ་བརྒྱ་ཐུབ་པའི་གོས་ཡོད་ཀྱང༌། །ཤི་བའི་ནངས་པར་གཅེར་བུ་འགྲོ།

བློ་ལྡན་མཁས་མཛངས་གང་ཡང་རུང༌། །དགེ་བཅུ་དཀའ་ཡང་མ་བཏང་ཞིག །མི་དགེ་བཅུ་སྦྱོང་སླ་ན་ཡང༌། །ངན་སོང་བཟོད་རླག་མེད་ལགས་པས། དེ་ལ་སོམས་ལ་ལྷ་ཆོས་མཛོད། །དང་པོར་སྐྱེས་ཀྱང་གཅིག་པུར་སྐྱེས། །ཐ་མར་འཆི་ཡང་གཅིག་པུར་འཆི། །དེ་ཕྱིར་དགེ་བའི་བཤེས་གཉེན་བཙལ། །ལེགས་པ་ཐམས་ཅད་སྟོན་པའི་ཕྱིར། །དགེ་བའི་བཤེས་གཉེན་ཕ་དང་འདྲ། །ཉེས་པ་ཐམས་ཅད་ཟློག་པའི་ཕྱིར། །དགེ་བའི་བཤས་གཉེན་མ་དང་འདྲ། མ་རིག་མུན་པ་སེལ་བའི་ཕྱིར། །དགེ་བའི་བཤེས་གཉེན་སྣང་གསལ་འདྲ། །ཉོན་མོངས་འདམ་ནས་འདྲེན་པའི་ཕྱིར། །དགེ་བའི་བཤེས་གཉེན་གྱད་དང་འདྲ། དེ་ཕྱིར་དགེ་བའི་བཤེས་གཉེན་བཙལ། །དམ་པའི་ཆོས་སྒོར་མ་ཞུགས་ན། །མི་ལུས་ཐོབ་ཀྱང་དོན་མེད་ཡིན། །ཟིན་མེད་ལས་ལ་མ་བརྩོན་པར། །བརྩོན་འགྲུས་ཇི་ཙམ་བྱས་གྱུར་ཀྱང༌། །ལས་གཞུག་མི་ཆོད་སྤྱོད་པ་ལ། །གྲུབ་པའི་སྙིང་པོ་མེད་ཀྱི། བྱར་མེད་སྟེང་དུ་སྟོང་དང༌། །འཛད་མེད་ལོང་གཏམ་སྨྲ་བ། །སྨྲ་བརྗོད་ཅི་ཙམ་བྱས་ཀྱང༌། །ཟིན་མེད་འཛད་པ་མེད་ལགས། །དགོས་པའི་དོན་གོ་ཆུང་གི །བརྗོད་མེད་ཉམས་སུ་ལོང་དང༌། །འགྲུབ་མེད་ཡིད་སྨོན་མ་གཏོང༌། །ཡིད་ཡུལ་ཙམ་བྱས་བསྲེགས་པ། བསམ་པའི་རྒྱང་ཚད་ཡིན་ལགས། །ཅི་བསམ་འགྲུབ་པར་དཀའ་ཡི། །གང་བྱུང་ལས་སུ་ཁྱེར་དང༌། །འབྲེལ་མེད་བརྩེ་གདུང་མ་ཆེ། བརྩེ་གདུང་ཇི་ཙམ་བསྐྱེད་ཀྱང༌། །རྟག་ཏུ་འགྲོགས་པ་མེད་ལགས། །རྨི་ལམ་རླུང་མ་འདྲ་ཡིས། །འབྲེལ་མེད་བློ་ཐག་ཆོད་ཅིག །འཕགས་པ་ལྗོན་ཤིང་གི་མདོ་རྫོགས་སོ།། །།
