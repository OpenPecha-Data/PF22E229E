༄༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྻ་ག་ན་པ་ཏི་ཧྲི་དཱ་ཡཱ། བོད་སྐད་དུ། འཕགས་པ་ཚོགས་ཀྱི་བདག་པོའི་གཟུངས། སངས་རྒྱས་དང༌། བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །འདི་སྐད་བདག་གིས་ཐོས་པ་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་རྒྱལ་པོའི་ཁབ། བྱ་རྒོད་ཕུང་པོའི་རི་ལ། དགེ་སློང་གི་དགེ་འདུན་ཆེན་པོ་རྣམས་དང་ཐབས་གཅིག་ཏུ་བཞུགས་སོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས། ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་པ། ཚོགས་ཀྱི་བདག་པོའི་སྙིང་པོ་འདི་འཛིན་ཅིང༌། ཀློག་པའི་མི་དེའི་ལས་ཐམས་ཅད་འགྲུབ་པར་འགྱུར་རོ། །ཡིད་ལ་འདོད་པ་ཐམས་ཅད་ཀྱང་འགྲུབ་པར་འགྱུར་རོ། །གསང་སྔགས་ཐམས་ཅད་ཀྱང་འགྲུབ་པར་འགྱུར་རོ། །ནོར་དང་ལོངས་སྤྱོད་ཐམས་ཅད་ཕུན་སུམ་ཚོགས་པར་འགྱུར་རོ། །མ་བླང་མ་བཙལ་བར་ཡང་རང་ཅི་འདོད་པའི་ཟས་ནོར་སྟེར་བར་འགྱུར་རོ། །དེའི་གཟུངས་འདི་བཟླས་པར་བྱའོ། །ཏདྱ་ཐཱ། ན་མོ་སྟུད་དེ། ག་ན་པ་ཏི། ཀ་ཊ་ཀ་ཊ། ཀི་ཊི་ཀི་ཊི། ཀུ་ཊ་ཀུ་ཊ། མ་ཏྲ་མ་ཏྲ། ད་ར་ད་ར། དྷ་ཧ་དྷ་ཧ། གྷྲི་ན་གྷྲི་ན། ད་བཱ་ད་བཱ། ཛམྦྷ་ཛམྦྷ། ས་མ་ཡ་མ་ནུ་སྨ་ར་ན་ཏུད་རེ་ཏུད་ཏྲ། པ་ཙ་ན་ཡེ་སྭཱ་ཧཱ། ཨ་བུ་ཏེ་བྷི་དུ་ཀྵ་བ་ནུ་ཙ་ཏ་ན། བ་ས་མ་ག་ར་ཚ། ཐ་མ་ཧ་བྷ་ཡ། མ་ཧཱ་བ་ཡཱ། མ་ཧེ་ཏེ་ཏ་ཀྱི་ནི་ཡ། པྲ་ཀོམ་པ་ཡ་སི། ཏདྱ་ཐཱ། ཨོཾ་ཀུ་རུ་ཀུ་རུ། མུ་རུ་མུ་རུ། ཙུ་རུ་ཙ་རུ། ན་མ་ན་མ་སྭཱ་ཧཱ། ཀུན་དགའ་བོ་རིགས་ཀྱི་བུའམ། རིགས་ཀྱི་བུ་མོའམ། དགེ་སློང་ཕ་དང༌། དགེ་སློང་མ་དང༌། དགེ་བསྙེན་ཕ་དང༌། དགེ་བསྙེན་མ་གང་ལ་ལ་ཞིག་གིས་ཚོགས་བདག་དབང་ཕྱུག་གི་སྙིང་པོ་འདི་ཉིན་རེ་བཞིན་དུ་བཏོན་ན། མི་དེ་ནི་ཚེ་འདི་ལ་བདེ་སྐྱིད་དང་ལྡན་པར་འགྱུར་རོ། །ཕྱུག་ཅིང་ལོངས་སྤྱོད་དང་ལྡན་པར་འགྱུར་རོ། །དབུལ་བ་དང་ཕོངས་པའི་སྡུག་བསྔལ་མི་འབྱུང༌། མི་ཀུན་གྱི་ཡིད་དུ་འོང་བར་འགྱུར་རོ། །ལས་ཅི་དང་ཅི་བསམས་པ་ཐམས་ཅད་ཀྱང་འགྲུབ་པར་འགྱུར་རོ། །ཚེ་འདི་དང་ཕྱི་མ་གཉིས་ཀ་ལ་ཡང་ནོར་དང་ལོངས་སྤྱོད་ཀྱིས་ཕོངས་པར་མི་འགྱུར་རོ། །ཉིན་གཅིག་བཞིན་དུ་ནངས་པར་སྔ་བར་ལངས་ནས།

གཟུངས་འདི་ལན་གསུམ་མམ་བདུན་བཏོན་ན། ཐོས་པ་འཛིན་པར་འགྱུར་རོ། །གནོད་སྦྱིན་དང༌། སྲིན་པོ་དང༌། ཤ་ཟ་དང༌། མཁའ་འགྲོ་མ་དང༌། དེ་དག་ཐམས་ཅད་ཀྱིས་མདངས་འཕྲོག་པར་མི་འགྱུར་རོ། །མི་དེ་ལ་བསྲུང་བ་དང༌། བསྐྱབ་པར་འགྱུར་རོ། །བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་བསྩལ་ནས། ཐམས་ཅད་དང་ལྡན་པའི་འཁོར་དེ་དག་དང༌། ལྷ་དང༌། མི་དང༌། ལྷ་མ་ཡིན་དང༌། དྲི་ཟར་བཅས་པའི་འཇིག་རྟེན་ཡི་རངས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལ་མངོན་པར་བསྟོད་དོ། །འཕགས་པ་ཚོགས་ཀྱི་བདག་པོའི་གཟུངས་རྫོགས་སོ།། །།མངྒ་ལཾ།། །།དེ་ནས་ཚོགས་ཀྱི་བདག་པོའི་སྤྲུལ་པ་སྤྲེའུ་ཡི་གཟུངས་འདི་ཅི་མང་དུ་བཟླས་ན། ཟས་དང་ནོར་གྱིས་ཕོངས་པར་མི་འགྱུར་རོ། །ཨོཾ་ན་མོ་ག་ན་པ་ཏི། ཛ་ཡ་བ་ར་ནེ། ཛི་ཝན་ཏི་ཡེ་སྭཱ་ཧཱ། །སྙིང་པོ་འདི་དུས་རྟག་ཏུ་བཟླས་བརྗོད་བྱས་ན། དུས་དེ་ཉིད་ལ་ནོར་གྱི་གཏེར་ཆེན་པོ་རྙེད་པར་འགྱུར་རོ། །སྟོང་འཁོར་ལོས་སྒྱུར་བའི་རྒྱལ་པོ་དང་སྐལ་བ་མཉམ་པར་འགྱུར་རོ། །སྔགས་ཀྱི་སྙིང་པོ་འདི་མག་དུ་བཟླས་ནས། རང་གི་མིག་ལ་བྱུགས་ན། མི་གང་མཐོང་བ་དེ་དབང་དུ་འདུས་པར་འགྱུར་རོ། །ནུབ་མོ་ཉལ་ཁར་སྔགས་ཆུ་མིག་ལ་བྱུགས་ནས་ཉལ་ན། ཅི་བསམས་པ་རྨི་ལམ་དུ་མཐོང་བར་འགྱུར་རོ། །སྐར་མ་རྒྱལ་གྱི་ཉིན་པར་གསེར་ལ་སོགས་པའི་རིན་པོ་ཆེ་ལ་སྤྲེའུ་ཡི་གཟུགས་ཕོ་མཚན་གྱེན་དུ་སྒྲེང་བ། ཕྱག་ཐལ་མོ་སྦྱར་བའི་བར་དུ། སྤོས་ཀྱི་རེང་བུ་ཐོགས་པ་གཅིག་བཞེངས་ལ། དེའི་སྟེང་དུ། ག་ན་པ་ཏི་བསྐྱེད་དོ། །གཟུངས་འདི་བཟླས་ཤིང༌། དུས་གསུམ་དུ་ཨུ་ག་རུ་ནག་པོའི་དུ་བས་བདུག་ཅིང་མཆོད་ན། ནོར་ཐམས་ཅད་འཕེལ་བར་འགྱུར་རོ། །ཨོཾ་ན་ག་ལ་སྐམ་ན་ཛ་ཡེ་སྭཱ་ཧཱ། །གེགས་ཐམས་ཅད་ཀྱིས་ནོར་སྟེར་བར་འགྱུར་རོ། །མི་ཐམས་ཅད་ཀྱིས་རྒྱལ་པོ་བཞིན་དུ་མཆོད་པར་འགྱུར་རོ། །གནོད་སྦྱིན་རྣམས་ཀྱིས་གཏེར་ཁ་འབྱེད་ཅིང༌། སྐྱེ་བ་ཕྱི་མ་ལ་ཡང་ནོར་གྱི་གཏེར་ཆེན་པོ་རྙེད་པར་འགྱུར་རོ། །ག་ན་པ་ཏིའི་གཟུངས་རྫོགས་སོ།། །།སརྦ་མངྒ་ལཾ།

ཤུ་བྷཾ།། །།
