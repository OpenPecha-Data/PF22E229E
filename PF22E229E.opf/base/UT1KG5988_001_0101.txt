ཨོཾ་ཨ་མི་དྷ་ཡཱ་ཡ་སྭཱ་ཧཱ། ཕྱག་འཚལ་བདག་ཅག་ཀུ་མིན་ཞེས་པའི་སྐར་མ་ཡི། །གོ་འཕང་ལའོ་ཁྱོད་ནི་ཤར་ཕྱོགས་རྨད་བྱུང་བའི། །རིན་པོ་ཆེ་ཡི་འཇིག་རྟེན་དག་ཏུ་རྒྱལ་བ་ནི། །གསལ་བའི་དབྱངས་ཀྱི་དབང་པོ་ཞེས་བྱའི་སྤྲུལ་པའོ། །དེ་ཡི་གཟུངས་ནི་འདི་ཡིན་ནོ། །ན་མཿས་མནྟ་བུདྡྷ་ནཱན། ཨོཾ་ཚིན་དྷ་ཧཱུཾ་ནི་སྭཱ་ཧཱ། བདག་ཅག་ཕྱག་འཚལ་ལུ་སུན་ཞེས་པའི་སྐར་མ་ཡི། །གོ་འཕང་ལའོ་ཁྱོད་ནི་ཤར་ཕྱོགས་འཇིག་རྟེན་གང༌། །ཟླ་བ་ཉ་འདྲའི་ཞིང་དུ་གསེར་གྱི་ཁ་དོག་ནི། །ཡོངས་སུ་རྫོགས་པའི་སངས་རྒྱས་ཀྱི་ནི་སྤྲུལ་པ་འོ། །དེ་ཡི་གཟུངས་ནི་འདི་ཡིན་ནོ། །ན་མཿས་མནྟ་བུདྡྷ་ནཱན། ཨོཾ་ནི་མི་རི་ནི་ལ་ཀ་ཧ་ན་བར་མ་རུ་ཧུ་སུ་སྭཱ་ཧཱ། ཕྱག་འཚལ་བདག་ནི་ལྦུ་གུན་ཞེས་པའི་སྐར་མ་ཡི། །གོ་འཕང་ལའོ་ཁྱོད་ནི་ཤར་ཕྱོགས་འཇིག་རྟེན་གང༌། །འགྱོད་པ་དང་ནི་སྨྲེ་སྔགས་མེད་པར་རྒྱལ་བ་ནི། །ཀུན་ལས་འཕགས་པའི་བསོད་ནམས་གོ་འཕང་སྤྲུལ་པའོ། །དེ་ཡི་གཟུངས་ནི་འདི་ཡིན་ནོ། །ན་མཿས་མནྟ་བུདྡྷ་ནཱན། ཨོཾ་ཝེ་ཧ་སྭཱ་ཧཱ། ཕྱག་འཚལ་བདག་ནི་ལིམ་ཚིམ་ཞེས་པའི་སྐར་མ་ཡི། །གོ་འཕང་ལའོ་ཁྱོད་ནི་ཤར་ཕྱོགས་འཇིག་རྟེན་གང༌། །གཙང་མར་ལྡན་པའི་ཞིང་དུ་བློ་དང་ཤེས་རབ་ནི། །མཁྱེན་པ་ཐོགས་མེད་རྒྱས་པ་རྒྱལ་བའི་སྤྲུལ་པའོ། །དེ་ཡི་གཟུངས་ནི་འདི་ཡིན་ནོ། །ན་མཿས་མནྟ་བུདྡྷ་ནཱན། །ཨོཾ་པ་ར་ཏི་ཙ་ར་མ་ར་ཎ་སྭཱ་ཧཱ། ཕྱག་འཚལ་བདག་ནི་ཧུ་གུ་ཞེས་པའི་སྐར་མ་ཡི། །གོ་འཕང་ལའོ་ཁྱོད་ནི་ཤར་ཕྱོགས་འཇིག་རྟེན་གང༌། །ཆོས་ལ་མོས་པར་ཆོས་ཀྱི་རྒྱ་མཚོར་དགྱེས་པ་ཡི། །ཐུབ་པ་དེ་བཞིན་གཤེགས་པའི་སྤྲུལ་པ་ཁྱོད་ཡིན་ནོ། །དེ་ཡི་སྔགས་ནི་འདིའོ། །ན་མཿས་མནྟ་བུདྡྷ་ནཱན། ཨོཾ་སརྦ་དྷ་ར་ས་མ་ཡེ་སྭཱ་ཧཱ། ཕྱག་འཚལ་བདག་ནི་ཕོ་གུན་ཞེས་པའི་སྐར་མ་ཡི། །གོ་འཕང་ལའོ་ཁྱོད་ནི་ཤར་ཕྱོགས་འཇིག་རྟེན་གང༌། །རིན་ཆེན་བཻ་ཌཱུརྻ་ལས་གྲུབ་པའི་ཞིང་དུ་ནི། །བཻ་ཌཱུརྻ་འོད་སྨན་པའི་རྒྱལ་པོའི་སྤྲུལ་པའོ། །དེ་ཡི་སྔགས་ནི་འདིའོ། །ན་མཿས་མནྟ་བུདྡྷ་ནན།

ཨོཾ་ས་པ་ཏ་པ་ར་མི་ན་ཙ་སྭཱ་ཧཱ། གང་སུ་དག་རང་གི་ལོའི་འགྲོས་ལ་བར་ཆད་བྱུང་བར་གྱུར་ན། དེ་ནས་མི་དེས་ཆོས་རིན་པོ་ཆེ་འདི་ལ་ལན་བདུན་དུ་ཕྱག་འཚལ་ལོ། །བར་ཆད་ཐམས་ཅད་ཞི་ཞིང་བདེ་བར་འགྱུར་རོ། །དེའི་ཚེ་ལྷའི་ལྷ་སངས་རྒྱས་ཀྱིས་བྱང་ཆུབ་སེམས་དཔའ་འཇམ་དཔལ་ལ་བཀའ་སྩལ་པ། འཇམ་དཔལ་ངས་བཤད་པའི་ཆོས་ཀྱི་མདོ་འདི་ནི་ཆོས་ཀྱི་མཆོག་དང་ལྡན་པ། འོད་དུ་གྱུར་པ། སྟོབས་དང་ལྡན་པའོ། །སེམས་ཅན་ཐམས་ཅད་ཀྱི་ནོར་བའི་ཉེས་པ་ལས་སྐྱོབ་པའི་རྟེན་པར་གྱུར་པའོ། །སྡིག་པ་དང་སྒྲིབ་པ་ཐམས་ཅད་དང༌། གདོན་གྱི་ལན་ཆགས་ཐམས་ཅད་གདེངས་པར་བྱེད་པ་དང་ཞི་བར་བྱེད་པའོ། །དགེ་སློང་ཕ་འམ། །དགེ་སློང་མ་འམ། སྦྱིན་བདག་ཕོ་མོའམ། ཁྱིམ་བདག་ཕོ་མོའམ། མི་དང་གྲགས་པ་དང་ལྡན་པའི་དམ་པར་གྱུར་བའི་སྐྱེས་པའམ། བུད་མེད་དམ། བཀུར་བར་འོས་པ་དང༌། བཀུར་བར་འོས་པ་མ་ཡིན་པ་ཆེན་པོ་དང་ཆུང་ངུ་གང་སུ་དག་སྐར་མ་སྨེ་བདུན་འདིའི་རིས་སུ་སྐྱེས་པའོ། །གང་སུ་དག་ཆོས་འདི་ཐོས་ནས་སློབ་པ་དང༌། ཁོང་དུ་ཆུད་ཅིང་རང་གི་ལུས་ཀྱིས་མཆོད་པ་དང༌། རིམ་པར་ཡང་གྲོགས་སམ། གཉེན་འདུན་དག་ལ་བསྐུལ་ནས་འཛིན་དུ་བཅུག་ན། བསོད་ནམས་འདིའི་དྲིན་དང་འབྲས་བུ་ཚེ་འདི་ཉིད་ལ་རྙེད་པར་འགྱུར་རོ། །རིགས་ཀྱི་བུའམ་རིགས་ཀྱི་བུ་མོ་གང་ལ་ལས། དམྱལ་བར་སྐྱེས་ནས་སྡུག་བསྔལ་སྣ་ཚོགས་རྩུབ་མོ་མྱོང་བར་གྱུར་པ་དག་གི་དོན་དུ་ཆོས་འདི་ལ་ཡི་རང་བའི་བསམ་པས། བཀུར་སྟི་དང་མཆོད་པ་བྱས་ན། དེ་ནས་དེ་དག་གི་ཤེས་པར་དམྱལ་བའི་སྡུག་བསྔལ་ལས་གྲོལ་ཞིང་སྲབ་པར་གྱུར་ནས། བདེ་བ་ཅན་གྱི་འཇིག་རྟེན་དུ་འོད་དཔག་ཏུ་མེད་པའི་སངས་རྒྱས་ཀྱི་ཞིང་དུ་སྐྱེ་བར་འགྱུར་རོ། །རིགས་ཀྱི་བུའམ་རིགས་ཀྱི་བུ་མོ་གང་སུ་དག་གདོན་དང་འབྱུང་པོ་དག་གིས་ཟིལ་གྱིས་གནོན་པའམ། བདུད་དྲག་པོས་སྡུག་བསྔལ་བར་བྱས་པའམ། །རྨི་ལམ་ངན་པ་རྨིས་པའམ། ལྟས་ངན་པ་མཐོང་བར་གྱུར་ཏམ།

རང་གི་སེམས་དངངས་སྐྲག་པར་གྱུར་ན། ཆོས་འདི་མཉན་ནས་བསླབས་ཤིང༌། ཁོང་དུ་ཆུད་པར་བྱས་ཏེ་མཆོད་ན། བར་ཆད་དེ་དག་ལས་གྲོལ་ཞིང་སྲབ་ནས། དེའི་བློ་དང་སེམས་བདེ་བར་འགྱུར། དེའི་སྐྲག་པ་དང་དངངས་པ་ཐམས་ཅད་དུ་ཟད་པར་འགྱུར་རོ། །རིགས་ཀྱི་བུའམ་རིགས་ཀྱི་བུ་མོ་གང་སུ་དག་ཇི་ལྟར་རང་གི་རྗེ་དཔོན་དང་ལྷན་ཅིག་ཏུ་འགྲོགས་ནས། མཉེས་ཤིང་གཅེས་པར་འགྱུར་བར་འདོད་ནས་སྤྱོད་པའི་དུས་སུའམ། ཡང་མངག་གཞུག་བྱེད་པར་གྱུར་ན་མི་དེ་ཆོས་འདི་དང་ཕྲད་ནས་རྗེས་སུ་ཡི་རང་བའི་སེམས་ཀྱིས་བཀུར་ཞིང་མཆོད་ན། དེ་ནས་རྗེ་དཔོན་དེ་ལ་གཉེན་ཞིང་གཅེས་པར་དེའི་མི་དང་གྲགས་པ་སྤེལ་ཞིང་གྲགས་པར་གྱུར་ནས། དགའ་བ་ཆེན་པོ་འཐོབ་པར་འགྱུར་རོ། །རིགས་ཀྱི་བུའམ་རིགས་ཀྱི་བུ་མོ་གང་སུ་དག་ནད་དྲག་པོས་ཐེབས་ནས་མནར་ཞིང༌། ནད་དེ་ལས་གྲོལ་བར་འདོད་པ་དག་གནས་གཙང་མར་སྤོས་བསྲེགས་ཤིང་ཆོས་འདི་ལ་མཆོད་པ་ཕུལ་ནས་ཀློག་ཏུ་བཅུག་ན། དེ་ལ་ཞེན་པའི་ནད་དེ་དག་ཐམས་ཅད་ལས་ཐར་བར་འགྱུར་རོ། །རིགས་ཀྱི་བུའམ་རིགས་ཀྱི་བུ་མོ་གང་སུ་དག་ལམ་དུ་ཞུགས་ཏེ་རིང་པོར་སོང་ནས་ཉོ་ཚོང་བྱེད་པ་དག་རྙེད་པའི་ནོར་གྱིས་ཚིམ་པར་འདོད་ནས། །དྲིན་དང་འབྲས་བུ་ཕན་ཡོན་ཡངས་ཤིང་རྒྱས་པར་འདོད་ན། ཆོས་འདི་ལ་ཡིད་ཆེས་པའི་བསམ་པས་བཀུར་ཞིང་མཆོད་ན། དེའི་འདོད་དོན་གང་ཡིན་པ་བསམ་པ་བཞིན་དུ་ཚིམ་པར་འགྱུར་རོ། །ཕྱི་རོལ་ཏུ་འགྲོ་བ་དང་ནང་དུ་འཇུག་པའི་དུས་སུ་དགའ་བ་རྒྱ་ཆེན་པོ་དང་ལྡན་པར་འགྱུར་རོ། །ཡང་རིགས་ཀྱི་བུའམ་རིགས་ཀྱི་བུ་མོ་གང་སུ་དག་ཞིང་ལ་ལོ་ཏོག་དང་ཤིང་ཐོག་གི་འབྲས་སུ་མ་རྙེད་པ་དང༌། ཕྱུགས་ཀྱི་ནད་བྱུང་བར་གྱུར་ན། དེ་ནས་གནས་གཙང་མར་སྤོས་བསྲེགས་ཏེ། ཆོས་རིན་པོ་ཆེ་འདི་ལ་མཆོད་ན་ལོ་ཏོག་ཐམས་ཅད་ལེགས་པར་འགྱུར། ཕྱུགས་འཕེལ་བར་འགྱུར་རོ། །ཡང་གདོན་དང་བར་ཆད་འབྱུང་བར་མི་འགྱུར་རོ། །ཡང་རིགས་ཀྱི་བུ་མོ་གང་ཞིག་མངལ་དུ་བུ་ཆགས་ནས་བར་ཆད་དང་ལྡན་པའི་ཟླ་བ་དང་འཛོམས་ན།

ཆོས་འདི་ལ་ཡིད་ཆེས་པའི་སེམས་ཀྱིས་བཀུར་ཞིང་མཆོད་ན། བར་ཆད་དེ་དག་ལས་ནད་མེད་པར་གྲོལ་བར་འགྱུར་རོ། །བཙས་པའི་བུ་དང་བུ་མོ་དེ་དག་ཀྱང་བསོད་ནམས་དང་བཀྲ་ཤིས་དང་ཚེ་རིང་བ་དང་ལྡན་པར་འགྱུར་རོ། །ཡང་རིགས་ཀྱི་བུ་དང་རིགས་ཀྱི་བུ་མོ་གང་སྐྱེས་པའི་མི་ཐམས་ཅད་སྨེ་བདུན་གྱི་སྐར་མས་ཤེས་ཤིང་བདག་གིར་བྱེད་པའོ། །ཇི་སྲིད་འཚོའི་བར་དུ་ཚེ་གཅིག་ལ་ཇི་ཙམ་དུ་བར་ཆད་དང་དོན་མེད་པའི་རྩོད་པ་དང༌། དེ་ལ་སོགས་པ་ལྟས་ངན་པ་བརྒྱ་ཕྲག་ཐམས་ཅད་སྨེ་བདུན་ཞེས་པའི་སྐར་མ་བདག་པོར་གྱུར་པའོ། །ཞེས་ཤེས་ནས་ཆོས་འདི་བཀུར་ཞིང་ཡིད་ཆེས་པར་བྱས་ནས་མཆོད་ན། གདོན་དང་བར་དུ་གཅོད་པ་གང་ཡང་གནོད་པ་ཁྱེད་མི་ནུས་སོ་ཞེས་བཤད་པ་ཡིན་ནོ། །དེ་ནས་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་སོགས་པ་འཁོར་བཞི་པོ་དེ་དག་འདུས་ཏེ། ཆོས་ཀྱི་བཀའ་འདི་བཀུར་ཞིང་བསྟི་སྟང་དུ་བྱས། ཀུན་ཆུབ་པར་བྱས་ནས་ཡན་ལགས་ལྔས་ཕྱག་བྱས་ཏེ་སོང་ངོ༌། །ན་མོ་རཏྣ་ཙིར་ཛ་མ་ཧཱ་ཏེ་ཛ་ཙིར་ཨ་བ་བྲ་ཨ་ཀི་སྭཱ་ཧཱ། ལྷའི་ལྷ་སངས་རྒྱས་ཀྱིས་གསུངས་པའི་སྨེ་བདུན་ཞེས་བྱ་བའི་སྐར་མས་ཚེ་རིང་པོར་བྱེད་པ་ཞེས་བྱ་བའི་ཆོས་ཀྱི་མདོའི་ལེའུའོ།། །།དཔལ་ལྡན་གསེར་གྱི་མི་ནི་གཟུགས་དཀར་པོ། །དཔལ་ལྡན་ཤིང་གི་མི་ནི་གཟུགས་སྔོན་པོ། །དཔལ་ལྡན་ཆུ་ཡི་མི་ནི་གཟུགས་ནག་པོ། །དཔལ་ལྡན་མེ་ཡི་མི་ནི་གཟུགས་དམར་པོ། །དཔལ་ལྡན་ས་ཡི་མི་ནི་གཟུགས་སེར་པོ། །སྨེ་བདུན་ཞེས་པའི་སྐར་མ་ལ་གསོལ་བ་འདེབས་པའི་བསྟོད་པ་ནི་འདི་ཡིན་ནོ།། །།ས་ཡི་སྙིང་པོའི་སྟེང་ལས་དཔག་ཚད་བརྒྱད་ཁྲི་འཕགས་པ་ལྷུན་པོའི་རྩེར་གནས་པ། །གླིང་བཞི་པ་ཡི་འཇིག་རྟེན་འདིར་ནི་དབང་དང་ལྡན་པ་ཁྱོད་ཉིད་བདག་གི་ལྷ། །སྐྱེ་གནས་བཞི་དང་ལས་ལྔ་དང་ལྡན་སེམས་ཅན་སྐྱབས་དང་སྐྱོབ་པ་ཁྱོད་ཡིན་ཏེ། །ཕྲེང་བ་བརྒྱུས་ལྟར་ལྷ་ཡི་རྒྱལ་པོ་བརྒྱ་བྱིན་གྱིས་ནི་དབུ་རྒྱན་གྱུར་པ་ཉིད།

ཉི་ཟླ་གཟའ་བདུན་རྒྱུ་སྐར་རྣམས་དང་དུང་ཕྱུར་ཁྲི་ཁྲག་སུམ་ཅུའི་སྐར་མས་མཆོད། །བདག་གི་ལྷ་ཁྱོད་ཚེ་རབས་འདིར་ནི་འདོད་པའི་དོན་ཀུན་འགྲུབ་པར་བྱེད་པ་སྟེ། །དེས་ནི་གང་འདོད་དོན་ཀུན་བསམས་དང་ཇི་ལྟར་འདོད་བཞིན་བདག་ཡིད་ཚམ་པར་མཛོད། གདོན་དང་བར་ཆད་བརྒྱ་ཕྲག་ཞི་ཞིང་མེད་པར་མཛད་པ་ཁྱོད་ཡང་བདག་གི་ལྷ། །ཚེ་རིང་བར་ཡང་མཛད་པ་ཁྱེད་ལགས་བདག་གི་བཟང་པོའི་ཚིག་ཀུན་འགྲུབ་པར་མཛོད། །སྨེ་བདུན་ཞེས་པའི་སྐར་མ་ལ་མར་མེ་འབུལ་བའི་ཉི་མ་དག་བཤད་པར་བྱའོ། །དཔྱིད་ཟླ་ར་བའི་ཚེས་བདུན་དང་དཔྱིད་ཟླ་འབྲིང་པོའི་ཚེས་བཞི་དང༌། དཔྱི་ཟླ་ཐ་མའི་ཚེས་གཉིས་དང༌། དབྱར་ཟླ་ར་བའི་ཉི་ཤུ་བདུན་དང༌། དབྱར་ཟླ་འབྲིང་པོའི་ཚེས་ལྔ་དང༌། དབྱར་ཟླ་ཐ་མའི་ཉི་ཤུ་གསུམ་དང༌། སྟོན་ཟླ་དང་པོའི་ཚེས་ཉི་ཤུ་དང༌། སྟོན་ཟླ་འབྲིང་པོའི་ཚེས་བཅུ་བདུན་དང༌། སྟོན་ཟླ་ཐ་མའི་ཚེས་ཉི་ཤུ་དང༌། དགུན་ཟླ་ར་བའི་ཚེས་བཅུ་གཅིག་དང༌། དགུན་ཟླ་འབྲིང་པོའི་ཚེས་བཅོ་ལྔ་དང༌། དགུན་ཟླ་ཐ་མའི་ཚེས་བརྒྱད་རྣམས་ལ་མར་མེ་འབུལ་བར་བྱའོ། །སྨེ་བདུན་ཞེས་པའི་སྐར་མའི་མདོ་རྫོགས་སོ།། །།སྟོན་པ་རྫོགས་པའི་སངས་རྒྱས་ཀྱིས་གསུངས་པའི། །སྨེ་བདུན་ཞེས་པའི་སྐར་མའི་མདོ་སྡེ་འདི། །བསྟན་པའི་སེམས་ཀྱིས་དྲན་ཞིང་གང་མཆོད་པ། །དེ་ལ་ཕན་པ་འབྱུང་ཞེས་རབ་ཤེས་ནས། །ཨུ་རུག་བོ་གའི་མིང་ཅན་ཟུ་གུར་ཆེ། །ཆུང་ངུའི་དུས་ནས་རྟག་པར་ཆོས་འདི་ལ། ཡིད་ཆེས་ལྡན་པས་རྒྱུན་དུ་ཀློག་ཅིང་མཆོད། །རང་གི་གོ་འཕང་འཚོལ་ཞིང་གསོལ་འདེབས་པས། །མཐུན་པར་སྐྱོབ་པའི་བདག་པོ་བསོད་ནམས་ཅན། །གྲོལ་མཛད་སྟོན་པ་སངས་རྒྱས་སྤྲུལ་པ་གང༌། །ཐེ་མུར་རྒྱལ་བུ་ཡུན་དུ་ཚེ་རིང་ཞང༌། །ཐུག་ཅན་རྒྱལ་བུ་ཆེན་པོར་འགྱུར་བར་འདོད། །བློ་ལྡན་བྱང་ཆུབ་སེམས་དཔའ་བདག་པོ་དེ། །སླེབས་ནས་སེ་ཆེན་རྒྱལ་པོའི་གདན་སར་བཞུགས། །བདག་གི་ཡིད་ལ་འདོད་པ་ཚིམ་གྱུར་པས། །ཐེ་ཚོམ་མེད་པར་ཆོས་འདིར་ངེས་ཤེས་སྐྱེས། །ཡུ་གུར་ཡི་གེར་ཆོས་ཀྱི་མདོ་སྡེ་འདི། །སྔོན་ཆད་གཞན་གྱིས་བསྒྱུར་བ་མེད་པས་ན། །མང་པོའི་ཧོར་རྣམས་དད་པས་མཆོད་གྱུར་ཅེས། །བདག་གིས་ཧོར་གྱིས་སྐད་དུ་བསྒྱུར་བ་ཡིན། །ང་ཡི་ཇི་ལྟར་བསམ་པ་འགྲུབ་འགྱུར་ན། །སེམས་ཅན་ཁྲི་ཁྲག་སྟོང་ཕྲག་འདོད་པ་ཀུན། །བདག་ཉིད་ཇི་བཞིན་ཚིམ་པར་གྱུར་ཅིག་ཅེས། །སྟོང་ཕྲག་སྤར་དུ་བཏབ་ནས་ཀུན་ལ་བཀྱེ། །འདི་ཡི་བསོད་ནམས་དྲིན་གྱི་འབྲས་བུའི་མཐུས། །བདག་པོ་རྒྱལ་པོ་དཔོན་མོ་ར་བརྒྱུད་པར་བཅས། །ཡུན་དུ་སྐུ་ཚེ་རིང་ཞིང་བསོད་ནམས་འཕེལ། །མཐར་ཐུག་སངས་རྒྱས་གོ་འཕང་རྙེད་གྱུར་ཅིག

།རྒྱལ་ཁམས་དགྲ་དང་འཁྲུག་པ་ཞི་བ་དང༌། །གདོན་དང་བར་ཆད་རིམས་མེད་བདེ་གྱུར་ཅིག །ཆར་རླུང་དུས་བབས་མུ་གེ་མེད་འགྱུར་ཞིང༌། །བདག་གིས་སྨྲས་དང་བསམ་དོན་འགྲུབ་གྱུར་ཅིག །བདག་དང་ཕ་མ་བུ་སོགས་གཉེན་རྣམས་དང༌། །འཚོ་བའི་སྤུན་དང་སེམས་ཅན་མང་པོར་བཅས། །འཇིག་རྟེན་འདིར་ཡར་ཆོས་ཀྱིས་འདོད་པ་ཚིམ། །བདེ་བ་ཅན་ཞིང་རྟག་པར་ཕྱིན་གྱུར་ཅིག །ཐི་ལི་དང་པོའི་ལོ་འབྲུག་གི་ཟླ་བ་བཅུ་པའི་ཚེས་གཅིག་ལ་སྤར་དུ་བཏབ་པ་ཡིན། མདོ་འདི་རྒྱ་གར་གྱི་ཡུལ་ནས་རྒྱ་གར་གྱི་པནྜི་ཏ་གཅིག་དང༌། ཐང་ཟམ་ཚང་གིས་ཁྱེར་ཏེ་རྒྱའི་ཡུལ་དུ་བསྒྱུར་རོ། །རྒྱ་ཆེན་པོའི་ཡུལ་དུ་རྒྱས་པར་གྱུར་ཅིང་གནས་པ་ལས་རྒྱལ་པོ་ཆེན་པོའི་བློན་པོ་བྱང་ཆུབ་སེམས་དཔའི་རིགས་སུ་བྱུང་བ་དད་པ་དང༌། ཤེས་རབ་དང༌། ཏིང་ངེ་འཛིན་དང་ལྡན་པ་གི་མ་རྩེ་གོང་ལུ་ཏའི་ཧུའི་གཡུའི་ཤྲི་ཐའི་ཨུ་རུག་པོ་ཡིས་བསྐུལ་ཏེ་ཡུ་གུར་གྱི་བསྟན་པའི་བདག་པོ་པྲཛྙཱ་ཤྲཱིས་ཧོར་གྱི་སྐད་དང་ཡི་གེར་བསྒྱུར་ནས་སྟོང་ཕྲག་གཉིས་སྤར་དུ་བཏབ། ཨ་ལི་ན་ཐི་མུར་ཏའི་སེ་དུས་ཡུ་གུར་གྱི་སྐད་དུ་བསྒྱུར་ཏེ། སྟོང་ཕྲག་གཅིག་སྤར་དུ་བཏབ་ནས། ཆོས་ཀྱི་སྦྱིན་པ་བྱས་ཏེ་ཧོར་དང་ཡུ་གུར་ལ་རྒྱས་པར་བྱས་ཤིང༌། ཏའི་ཧུ་ཉིད་ཀྱང་སྔོན་ཆད་སོག་པོའི་ཆོས་ལུགས་འཛིན་པ་ལ། འདིའི་བྱིན་རླབས་གྱིས་སངས་རྒྱས་ཀྱི་ཆོས་ལ་བཞུགས་ཤིང་རབ་ཏུ་བྱུང་སྟེ་འདིའི་ཡོན་ཏན་རྣམས་ཉམས་སུ་མྱོང་བར་འགྱུར་རོ། །སླད་ཀྱི་མེ་མོ་གླང་གི་ལོ་ལ་ལོཙྪ་བ་མ་ཧཱ་ཕ་ལ་དང༌། ཤྲཱི་ཨཱ་ནནྟ་བཛྲས་གུང་ཐང་གི་གཙུག་ལག་ཁང་དུ་བོད་ཀྱི་སྐད་དང་ཡི་གེར་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ།། །།
