༄༅། །རྒྱ་གར་སྐད་དུ། བཛྲ་ཧྲི་པ་ཤ་ཤ་གུ་ཡ། བོད་སྐད་དུ། གཟའི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་བྱེད་པའི་གཟུངས། རྡོ་རྗེ་མི་བསྐྱོད་པ་ལ་ཕྱག་འཚལ་ལོ། །འདི་སྐད་བདག་གིས་ཐོས་པ་དུས་གཅིག་ན། དཔལ་ཕྱག་ན་རྡོ་རྗེ་བསིལ་བའི་ཚལ་དུ་བཞུགས་སོ། །དེའི་ཚེ་གཟའ་དང་རྒྱུ་སྐར་དབང་དུ་སྡུད་པའི་གཟུངས་འདི་གསུངས་སོ། །ཏདྱ་ཐཱ། ཨོཾ་བཛྲ་པཱ་ཎི་ཧཱུཾ། ཧ་ས་ཧ་ས། དྷ་ར་དྷ་ར་སྭཱ་ཧཱ། དེ་སྐད་ཅེས་གསུངས་པ་དང༌། ཡབ་སྲིན་པོ་དང༌། ཡུམ་ཀླུ་མོ་དང༌། དྲང་སྲོང་ཆེན་པོ་རཱ་ཧུ་ལ་དང༌། སྲིན་མོ་གདོང་མོ་བཞི་དང༌། རྒྱུ་བའི་གཟའ་བདུན་དང༌། རྒྱུ་སྐར་ཉི་ཤུ་རྩ་བརྒྱད་དང༌། གེགས་དང༌། སྡེ་བརྒྱད་ལ་སོགས་པ་རྣམས། ཕྱག་ན་རྡོ་རྗེའི་འཁོར་དུ་འདུས་ཏེ་གུས་པར་བྱས་ནས། རང་རང་གི་གཟའི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་བྱེད་པའི་གཟུངས་འདི་གསུངས་སོ། །དེ་ནི་འདི་ལྟ་སྟེ། ཨོཾ་བཛྲ་ཡཀྵ་གྷན་དྷེ་ཧུ་ལུ་ཧུ་ལུ་ཧཱུཾ་ཕཊ། ཡབ་ཡག་ཤ་ནག་པོ་ཐོད་པའི་ཕྲེང་བ་ཅན་གྱི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ན་ག་རཱ་ཧུ་ལ་མུ་ལུ་མུ་ལུ་སྭཱ་ཧཱ། ཡུམ་ཧཾ་ཤཾ་སྦལ་མགོ་ཁྲག་མིག་མའི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་པ་ཤ་མི་ཀི་རི་ཙུ་ལུ་ཙུ་ལུ་སྭཱ་ཧཱ། དྲང་སྲོང་གཟའི་རྒྱལ་པོ་རཱ་ཧུ་ལའི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་རག་མོ་མ་མ་སྭཱ་ཧཱ། སྲིན་མོ་ཤར་གྱི་སྟག་སྲིན་ཟོར་བའི་གདོང་ཅན་གྱི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ། ཨོ་བཛྲ་རག་མོ་མུ་མུ་ས་ཧཱ། སྲིན་མོ་ལྷོའི་དམ་སྲི་དྲེད་མོའི་གདོང་ཅན་གྱི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་རག་མོ་མེ་མེ་སྭཱ་ཧཱ། སྲིན་མོ་ནུབ་ཀྱི་རྒོད་སྲིན་བ་གླང་གདོང་ཅན་གྱི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་ས་ཧཱ། ཨོཾ་བཛྲ་རག་མོ་མོ་མོ་སྭཱ་ཧཱ། སྲིན་མོ་བྱང་གི་འབྲོག་སྲིན་ཆུ་སྲིན་གདོང་ཅན་གྱི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ།

ནད་པ་འདི་ལ་བསྲུང་དུ་གསོལ་བསྲུང་དུ་གསོལ་ལོ། །དྲང་སྲོང་ཆེན་པོ་རཱ་ཧུ་ལའི་སྤྲུལ་པ་འགྱུ་བའི་གཟའ་བདུན་གྱི་སྙིང་པོ་འདི་ལྟ་སྟེ། ཨོཾ་བཛྲ་ཤ་ཤྀ་ར་དྷ་ར་དྷ་ར་སྭཱ་ཧཱ། གཟའ་ཉི་མའི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ། ཨ་བཛྲ་རྨུ་རྨུ་བཾ་དྷི་རི་དྷི་རི་སྭཱ་ཧཱ། གཟའ་ཟླ་བའི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་མེ་མེ་རི་དྷུ་རུ་དྷ་རུ་སྭཱ་ཧཱ། གཟའ་མིག་དམར་གྱི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་མུ་མུ་བཾ་དྷ་ར་དྷ་ར་སྭཱ་ཧཱ། གཟའ་ལྷག་པའི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་ས་ཧུ། ཨོཾ་ཤཾ་ཤཾ་ཤིང་དྷ་རོ་དྷ་རོ་སྭཱ་ཧཱ། གཟའ་ཕུར་བུའི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ། ཨོཾ་ལཾ་ལཾ་ཁ་དྷ་ཁ་དྷ་སྭཱ་ཧཱ། གཟའ་པ་ཝ་སངས་ཀྱི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ། ཨོཾ་ས་ས་དྷ་ས་དྷ་ས་སྭཱ་ཧཱ། གཟའ་སྤེན་པའི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བ་ར་གྱུར་ཅིག་སྭཱ་ཧཱ། ནད་པ་འདི་ལ་བསྲུང་དུ་གསོལ་བསྲུང་དུ་གསོལ་ལོ། །ཉི་ཟླ་གཟའི་ཕོ་ཉ་རྒྱུ་སྐར་ཉི་ཤུ་རྩ་བརྒྱད་ཀྱི་གཟུངས་ནི་འདི་ལྟ་སྟེ། ཨོཾ་བཛྲ་དུ་ཤ་ན་ཏི་ན་མ་གྷ་ན་མ་ཀ་ར་ལྷང་ལྷང་ཙེར་ཙེར་བ་གཏི་བ་ནི་སྭཱ་ཧཱ། རྒྱུ་སྐར་ཉི་ཤུ་རྩ་བརྒྱད་ཀྱི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ། དྲང་སྲོང་ཆེན་པོ་རཱ་ཧུ་ལའི་སྤྲུལ་པ་འགྱུ་བའི་གཟའ་བདུན་གྱི་གཟུངས་ནི་འདི་ལྟ་སྟེ། ཨོཾ་སྲིན་སྲིན་མུ་ཤ་མ་ཉ་ན་ཤེ་མ་ཤུ་ཀ་ཧ་རི་ཀུ་རི་མུ་རིསུ་རི་ཕུ་རི་དྷ་རི་མུ་སྭཱ་ཧཱ། ཀོང་བཙུན་མོ་དང༌། དགྲ་ལྷ་མོ་དང༌། མ་མོ་ལ་སོགས་ཏེ་སྡེ་བརྒྱད་ཀྱི་ནད་ཐམས་ཅད་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ། ནད་པ་འདི་ལ་བསྲུང་དུ་གསོལ་བསྲུང་དུ་གསོལ་ལོ། །དྲང་སྲོང་གཟའི་བྲན་དང་གཡོག་ཏུ་གྱུར་པའི་གེགས་རིགས་སྟོང་ཕྲག་བརྒྱད་ཅུའི་ནད་ཐམས་ཅད་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཤཾ་ཧ་ཏྲྀ་སུཾ་རུ་ཏྲ་མ་ཁ་ཏྲ་ཡེ་སྭཱ་ཧཱ། གེགས་རིགས་སྟོང་ཕྲག་བརྒྱད་ཅུའི་གནོད་པ་ཐམས་ཅད་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ།

ནད་པ་འདི་ལ་བསྲུང་དུ་གསོལ་བསྲུང་དུ་གསོལ་ལོ། །གཟའི་གདོན་ཐམས་ཅད་མི་གཙང་བ་ལས་གྱུར་པས། ཁྲུས་ཀྱི་གཟུངས་འདི་ལྟ་སྟེ། ཨོཾ་ཙནྜ་མ་ཧཱ་ཡཀྵ་བཛྲ་ཙནྜ་ཀྲོ་དྷ་ཡ། ཧུ་ལུ་ཧུ་ལུ། ཏིཥྛ་ཏིཥྛ། བྷན་དྷ་བྷན་དྷ། ཧ་ན་ཧ་ན། ཨ་མྲི་ཏེ་ཧཱུཾ་ཕཊ། མི་གཙང་བ་ཐམས་ཅད་དག་པར་གྱུར་ཅིག །མི་གཙང་བ་ཐམས་ཅད་ཞི་བར་གྱུར་ཅིག །གཟའ་དང༌། རྒྱུ་སྐར་གྱི་ནད་དང༌། བྱད་སྟེམས་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག་སྭཱ་ཧཱ། ནད་པ་འདི་ལ་བསྲུང་དུ་གསོལ་ལོ། བསྲུང་བར་མཛད་དུ་གསོལ་ལོ། །གཟའ་མི་རྒྱུ་བར་གྱུར་ཅིག །གཟའ་མི་ལྡོག་པར་གྱུར་ཅིག །གཟའ་མི་འབབ་པར་གྱུར་ཅིག །སྡེ་བརྒྱད་ཀྱི་ཆད་པ་ལས་ཐར་བར་གྱུར་ཅིག །གཟའ་གནས་སུ་བཞུགས་པར་གྱུར་ཅིག །གཟའི་ནད་ཐམས་ཅད་ལས་གྲོལ་བར་གྱུར་ཅིག །གཟུངས་སྔགས་འདིའི་མཐུས། འགྲོ་བ་མིའི་ལུས་དྲང་སྲོང་གཟའི་བྱད་ཀ་གྲོལ་བར་གྱུར་ཅིག །དེ་སྐད་ཅེས་བཀའ་བསྩལ་པ་དང༌། ཡབ་ཡུམ་མིང་སྲིང་དང༌། རྒྱུ་བའི་གཟའ་བདུན་དང༌། རྒྱུ་སྐར་ཉི་ཤུ་བརྒྱད་དང༌། གེགས་དང༌། སྡེ་བརྒྱད་ལ་སོགས་པས་རང་རང་གི་སྙིང་པོ་ཕུལ་བས། དཔལ་ཕྱག་ན་རྡོ་རྗེ་ཐུགས་དགྱེས་ཏེ། དམ་ལ་བཏགས་ནས། རང་རང་གི་གནས་སུ་བདེ་བ་ལ་བཀོད་དོ། །མངོན་པར་བསྟོད་དོ། །གཟའི་ནད་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་བྱེད་པའི་གཟུངས་ཀྱི་མདོ་དྲང་སྲོང་ཡབ་ཀྱི་གཟུངས་རྫོགས་སོ།། །།ཤུ་བྷ་མསྟུ་སརྦ་ཛཱ་ག་ཏྭཾ།། །།
