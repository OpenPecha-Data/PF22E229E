༄༅། །རྒྱ་གར་སྐད་དུ། ཡཀྵ་ཨ་པ་ར་བི་ཤུདྡྷ་ནཱ་མ་དྷ་ར་ཎི། བོད་སྐད་དུ། གནོད་སྦྱིན་གཞན་གྱིས་མི་ཐུབ་པ་ཡང་དག་ཤེས་ཀྱི་གཟུངས། དཔལ་ཕྱག་ན་རྡོ་རྗེ་ལ་ཕྱག་འཚལ་ལོ། །འདི་སྐད་བདག་གིས་ཐོས་པ་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་དཔལ་ཕྱག་ན་རྡོ་རྗེ། རི་རབ་རིའི་རྩེ་མོ་བྱང་ཤར་མཚམས་ཀྱི་ཕོ་བྲང༌། ངལ་གསོ་བའི་གནས་སུ་བཞུགས་པའི་ཚེ། གྲངས་མེད་པའི་ལྷ་དང༌། ཀླུ་དང༌། དྲི་ཟ་དང༌། ལྷ་མ་ཡིན་དང༌། ནམ་མཁའ་ལྡིང་དང༌། མི་འམ་ཅི་དང༌། ལྟོ་འཕྱེ་ཆེན་པོ་དང༌། འཇིག་རྟེན་སྐྱོང་བ་ཐམས་ཅད་དང༌། གནོད་སྦྱིན་ངལ་བསོ་བ་དང༌། ཡང་དག་ཤེས་དང༌། ནོར་བུ་བཟང་པོ་དང༌། གང་བ་བཟང་པོ་དང༌། ཛམ་བྷ་ལ་ལ་སོགས་པ། བཅོམ་ལྡན་འདས་ཀྱི་སྤྱན་སྔར་ལྷགས་ནས་འཁོད་དོ། །དེ་ནས་གནོད་སྦྱིན་གྱི་རྒྱལ་པོ་ཡང་དག་ཤེས་ཨ་པ་རའི་ཆ་ལུགས་འཛིན་པ། སྐུལ་ནོར་བུ་སྣ་ཚོགས་ཀྱིས་བརྒྱན་ཏེ། ཟླ་བ་དང་ཉི་མ་ལྟར་འབར་བའི་སྐུ་རབ་ཏུ་བཞད་པ་སྟན་ལས་ལངས་ཏེ། བཅོམ་ལྡན་འདས་ཕྱག་ན་རྡོ་རྗེ་ག་ལ་བ་དེར་སོང་སྟེ། བྷ་ག་ཝཱན་ལ་རིན་པོ་ཆེ་སྣ་བདུན་གྱི་མཆོད་པ་བྱས་ནས་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བྷ་ག་ཝཱན། བདག་གིས་བཅོམ་ལྡན་འདས་ཀྱི་ཞབས་ལ་དངོས་གྲུབ་ཐམས་ཅད་འབྱུང་བ་དང༌། ཡིད་བཞིན་གྱི་ནོར་བུ་ཞེས་བྱ་བ་འབུལ་གྱིས་བཞེས་སུ་གསོལ། ན་མོ་བུདྡྷ་ཡ། ན་མོ་དྷརྨ་ཡ། ན་མ་སངྒྷ་ཡ། ན་མ་པྲ་མ་ཡ། ན་མ་ཨིན་དྲཱ་ཡེ། ན་མོ་ཙ་ཏུ་ན་མ། མ་བྷེ་ར་ཤ་ན་མ། ཏདྱ་ཐཱ། ཧི་ལེ་ཧི་ལེ། མི་ལེ་མི་ལེ། གུ་རི་མ་ཏུ་རི། མ་ཧཱ་གྷ་དྷ་རི། དྷ་ར་མི་ཀྱེ། དྷ་ན་དྷ། ཁུ་ཁུན་དྷེ། ཧ་ཧ་ཧ་ཧ་ཧ། ཧི་ཧི་ཧི་ཧི་ཧི། ཧོ་ཧོ་ཧོ་ཧོ་ཧོ། ཧ་ལོ་དམ་བྷེ། ཙ་ཙ་ཙ་ཙ། ཙི་ཙི་ཙི་ཙི། ཙུ་ཙུ་ཙུ་ཙུ། ཤྩ་ན་དྷ་ཤྱལ། ཤི་གྲ་ཤི་གྲ། ཨུན་ཏི་ས་ཐ་ཏི། བྷ་ག་ཝཱན། སིང་ཤིང་དྷི་ལཱ་ཡ་སྭཱ་ཧཱ། འདིའི་ཆོ་ག་ནི་གནོད་སྦྱིན་ཡང་དག་ཤེས་ཀྱི་གཟུངས། འཕང་དུ་ཆ་བཞི་ལྔ་ཙམ་ཡོད་པ་ལ། གཡས་ན་དགྲ་སྟ་ཐོགས་པ། གཡོན་ནེའུ་ལེའི་ཁ་ནས་རིན་པོ་ཆེ་སྐྱུག་པ་ཐོགས་པ།

ལུས་སྒེག་པ། སྤྱན་དམར་བ། མཆེ་བ་གཙིགས་པ། སྐུ་མདོག་ནག་པོ་གཅིག་བྲིས་ལ་དེའི་མདུན་ན་ད་ཀྱིལ་འཁོར་གྲུ་བཞི་པ་གཅིག་བྱས་ལ། དེའི་ཕྱོགས་བཞིར་བུམ་པ་བཞི། དཀར་གསུམ་མངར་གསུམ་གྱི་ཆབ་ཀྱིས་བཀང་ལ་བཞག་གོ །བྱུག་པ་དང༌། ཕྱེ་མ་དང༌། བདུག་པ་དང༌། མེ་ཏོག་རྣམས་བཀྲམ་མོ། །དེའི་མདུན་དུ་ཐབ་གཅིག་བྱས་ལ། དེར་གཞན་ཤིང་ངམ། ཤུག་པའི་མེ་སྦར་ལ། གཟུངས་གོང་མ་བརྒྱ་རྩ་བརྒྱད་བཟླས་ལ་བསྲེག་བླུགས་རེ་རེ་འབུལ་བས་ནི། དེའི་དུས་སུ་ཡང་དག་ཤེས་འོང་སྟེ། ཁྱོད་ཅི་འདོད་ཅེས་ཟེར་དུ་འོང་ངོ༌། །སྒྲུབ་པ་པོ་བདག་ལ། གསེར་དང༌། དངུལ་དང༌། རིན་པོ་ཆེ་ཐམས་ཅད་དང༌། ས་འོག་གི་གཏེར་དང༌། བར་སྣང་གི་འགྲོ་ཤེས་པ་དང༌། མིག་གི་མངོན་པར་ཤེས་པ་དང༌། རྣ་བའི་མངོན་པར་ཤེས་པ་དང༌། གཞན་གྱི་སེམས་ཤེས་པ་དང༌། སེམས་ཅན་གྱི་ཡིད་འཕྲོག་པ་དང༌། འཚོ་བའི་ཡོ་བྱད་སྨན་པ་དང༌། བཟའ་བ་དང༌། བཏུང་བ་དང༌། བསྒོ་བ་དང༌། མེ་ཏོག་དང༌། འཕྲེང་བ་དང༌། རྒྱན་ལ་སོགས་པ་ཅི་འདོད་པ་ཐམས་ཅད་གསོལ་བ་གདབ་བོ། །དེས་ཐོབ་པར་གྱུར་ཏེ། ཐེ་ཚོམ་མེད་དོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ལག་ན་རྡོ་རྗེས། གནོད་སྦྱིན་གྱི་རྒྱལ་པོ་ཡང་དག་ཤེས་ལ་ལེགས་སོ་ཞེས་བྱ་བ་བྱིན་ནོ། །ཕྱི་མའི་དུས་སུ་བསྟན་པ་འཛིན་ཅིང་སྐྱོང་བ་ལ་ནོར་མེད་པ་དང༌། དད་པ་དང་བརྩོན་འགྲུས་ཆུང་བ་དང༌། བླ་མ་དང་དཀོན་མཆོག་མཆོད་པའི་ཡོ་བྱད་མེད་པ་རྣམས་ལ་ནོར་བྱིན་ཅིག །ཅེས་བཀའ་བསྩལ་པ་དང༌། དེ་ནས་འཁོར་ཐམས་ཅད་དགའ་མགུ་ཡི་རངས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལ་མངོན་པར་བསྟོད་དོ། །ཨ་པ་ར་ཡང་དག་ཤེས་ཀྱི་གཟུངས་རྫོགས་སོ།། །།སརྦ་མངྒ་ལཾ། ཞུས་དག །ཤུ་བྷཾ།། །།
