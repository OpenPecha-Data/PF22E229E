ཕ་ཁུའི་བྱུར་འདྲེ་དང༌། དགྲ་བོ་དང༌། ཉེ་དུའི་ལན་ཆགས་དང༌། ལོ་ངན་པའི་སྐེག་དང༌། ཟླ་བའི་སྐེག་དང༌། བཀྲ་མི་ཤིས་པ་དང༌། སྐྱེ་བོའི་རྩོད་པ་དང༌། རྒྱལ་པོའི་ཆད་པ་དང༌། མཚང་འབྲུ་བ་དང༌། ཁ་མཆུ་རྩོད་པ་དང༌། ངན་སེམས་དང༌། བསམ་པ་ངན་པའི་གྲོགས་དང༌། མིག་བལྟས་ངན་པ་དང༌། གདོན་ཐམས་ཅད་བཟློག་པར་འགྱུར་རོ། །སྡང་བ་རྣམས་འདུལ་དུ་གསོལ། ཁ་མཆུ་དང༌། བྱོལ་ཟོར་དང༌། བོད་གཏོང་དང༌། རིམས་དང༌། ལྟས་ངན་པ་ཐམས་ཅད་ཞི་བ་ཤཱནྟིཾ་ཀུ་རུ་ཡེ་སྭཱ་ཧཱ། རིགས་ཀྱི་བུ་ཁ་མཆུ་ནག་པོ་ཞི་བར་བྱེད་པ་ཞེས་བྱ་བའི་གཟུངས་འདི་ནི། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱིས་བྱིན་གྱིས་བརླབས་པའོ། །ངས་ཀྱང་དེ་ལྟར་བཤད་པར་བྱའོ། །འདི་རབ་ཏུ་བཟུང་ལ་རྒྱུན་དུ་ཐོན་ཅིག །འཇིགས་པ་ཐམས་ཅད་སེལ་བར་འགྱུར་རོ། །ཁ་མཆུ་དང་ལྟས་ངན་པ་ཐམས་ཅད་བཟློག་པར་འགྱུར་རོ། །ནད་དང་གདོན་དང༌། འབྱུང་པོ་དང༌། །ལོ་སྐེག་དང༌། ཟླ་སྐེག་དང༌། མོ་ངན་པ་ཐམས་ཅད་ཞི་བར་འགྱུར་རོ། །དེ་ནས་བྱང་ཆུབ་སེམས་དཔའ་འཇམ་དཔལ་སྟན་ལས་ལངས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱི་ཆོས་ཀྱི་རྣམ་གྲངས་འདིའི་མིང་ཅི་ལགས། འདི་ཇི་ལྟར་བཟུང་བར་བགྱི་ཞེས་གསོལ་པ་དང༌། བཀའ་སྩལ་པ། འཇམ་དཔལ་འདི་ནི་ཁ་མཆུ་ནག་པོ་ཞི་བར་བྱེད་པ་ཞེས་བྱ་བ་ཟུངས་ཤིག །དེ་ནས་འཇམ་དཔལ་གྱིས་གསོལ་པ། ལེགས་སོ་ལེགས་སོ། །བཅོམ་ལྡན་འདས་མདོ་སྡེ་ཟབ་མོ་འདི་འཆང་བ་དང་ཀློག་པའི་མི་དེ་ལ་བདག་གིས་བསྲུང་བར་བགྱིའོ། །བསྐྱབ་པར་བགྱིའོ། །སྨྲ་བ་ན་ཚིག་རྩུབ་ཐུབ་པར་བགྱིའོ། །བཀའ་སྩལ་པ། གང་ཞིག་གཞན་དང་ཁ་མཆུ་རྩོད་པའི་ཚེ། མདོ་འདི་ལན་བདུན་བཀླགས་ཏེ། འཕགས་པ་འཇམ་དཔལ་ཡིད་ལ་བསྒོམས་ཏེ་སོང་ན། ཕ་རོལ་གྱི་རྩོད་པ་ཐམས་ཅད་འཇོམས་སོ། །རང་གི་ཚིག་བཙུན་པར་འགྱུར་རོ། །སྨྲས་པ་ལ་གཞན་ཡིད་ཆེས་པར་འགྱུར་རོ། །བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་བསྩལ་ནས། ཁྱིམ་བདག་མཛེས་ལྡན་དང༌། འཁོར་ཐམས་ཅད་དང༌། ལྷ་དང༌། མི་དང༌།

ལྷ་མ་ཡིན་དང༌། དྲི་ཟར་བཅས་པའི་འཇིག་རྟེན་ཡི་རངས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ལ་མངོན་པར་བསྟོད་དོ། །འཕགས་པ་ཁ་མཆུ་ནག་པོ་ཞི་བར་བྱེད་པ་ཞེས་བྱ་བ་ཐེག་པ་ཆེན་པོའི་མདོ་རྫོགས་སོ།། །།
