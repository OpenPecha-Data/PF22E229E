༄༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྻ་བཛྲ་ལོ་ཧ་དུཎྜ་ནཱ་མ་དྷཱ་ར་ཎི། བོད་སྐད་དུ། འཕགས་པ་རྡོ་རྗེ་གནམ་ལྕགས་ཀྱི་མཆུ་ཞེས་བྱའི་གཟུངས། སངས་རྒྱས་དང༌། བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །འདི་སྐད་བདག་གིས་ཐོས་པ་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་ཡུལ་མ་ག་དྷ་ན་རྒྱུ་ཞིང་ཨ་མྲའི་ཚལ་དུ་གཤེགས་ཏེ་ཕྱིན་ནས། ཨ་མྲའི་ཚལ་དུ་ལུས་འཕགས་རིའི་དབང་པོའི་བྲག་ཕུག་ན་བཞུགས་སོ། །དེ་ནས་ལྷའི་དབང་པོ་བརྒྱ་བྱིན་གྱིས། བཅོམ་ལྡན་འདས་ག་ལ་བ་དེར་ཕྱིན་ནས། བཅོམ་ལྡན་འདས་ཀྱི་ཞབས་ལ་མགོ་བོས་ཕྱག་འཚལ་ཏེ། ཕྱོགས་གཅིག་ཏུ་འདུག་གོ །དེ་ནས་ཚངས་པ་དང༌། ཁྱབ་འཇུག་དང༌། དབང་ཕྱུག་དང༌། རྒྱལ་པོ་ཆེན་པོ་བཞིས་བཅོམ་ལྡན་འདས་ལ་ལན་གསུམ་བསྐོར་བ་བགྱིས་ནས། བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་བདག་ཅག་ཐམས་ཅད་འདུས་ནས། ཀླུ་ཐམས་ཅད་ཀྱི་སྙིང་འཕྲོག་པར་བགྱིད་པ། སྙིང་འགྱེས་པར་བགྱིད་པ། ཀླུ་མ་རུངས་པ་རྣམས་ཀྱི་རིགས་ཕུང་བར་འགྱུར་བ། རླུང་དང་སེར་བའི་སྤྲིན་དང༌། མ་རུངས་པ་ཐམས་ཅད་བཟློག་པར་བགྱིད་པ། མེ་ཏོག་དང༌། འབྲས་བུ་དང༌། ལོ་ཏོག་འབྱོར་པར་བགྱིད་པ། རྡོ་རྗེ་གནམ་ལྕགས་མཆུ་ཞེས་བྱ་བའི་གཟུངས་བརྗོད་པར་འཚལ་གྱིས། བཅོམ་ལྡན་འདས་ཀྱིས་སེམས་ཅན་ཐམས་ཅད་ཀྱི་དོན་མཛད་པའི་སླད་དུ་བྱིན་གྱིས་བརླབ་ཏུ་གསོལ། བཅོམ་ལྡན་འདས་ཀྱིས་ཚངས་པ་དང༌། ཁྱབ་འཇུག་དང༌། དབང་ཕྱུག་དང༌། རྒྱལ་པོ་ཆེན་པོ་བཞི་ལ་བཀའ་སྩལ་པ། རིགས་ཀྱི་བུ་རྣམས་ངས་སེམས་ཅན་ཐམས་ཅད་ཀྱི་དོན་དང༌། འཕགས་པ་ཐམས་ཅད་ཀྱི་བདེན་པའི་བྱིན་གྱིས་བརླབས་ཀྱིས་སྨྲོས་ཤིག་ཅེས་བཀའ་སྩལ་པ་དང༌། དེ་ནས་ཚངས་པ་དང༌། ཁྱབ་འཇུག་ཆེན་པོ་དང༌། དབང་ཕྱུག་ཆེན་པོ་དང༌། རྒྱལ་པོ་ཆེན་པོ་བཞི་ཡིས་གཟུངས་སྔགས་སྨྲས་སོ། །དཀོན་མཆོག་གསུམ་ལ་ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་ཤཱཀྱ་ཐུབ་པ་ལ་ཕྱག་འཚལ་ལོ། །དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །ཕ་རོལ་ཟིལ་གྱིས་གནོན་པའི་མཐུ་བོ་ཆེ་རྣམས་ལ་ཕྱག་འཚལ་ལོ། །འཇིག་རྟེན་གསུམ་གྱི་གཙོ་བོ་ལ་ཕྱག་འཚལ་ལོ། །རྒྱལ་པོ་ཆེན་པོ་བཞི་ལ་ཕྱག་འཚལ་ལོ། །རྡོ་རྗེ་གནམ་ལྕགས་མཆུ་འབར་བ་མ་རུངས་པ་དྲག་ཏུ་དགོད་པ་ལ་ཕྱག་འཚལ་ལོ། །མ་ཐ་མ་ཐ།

པྲ་མ་ཐཱ་པྲ་མ་ཐཱ། ཛྭ་ལི་ཏ་བྷི་ཀྲི་ཏ། བཛྲ་ཛྭ་ཡ་ཛྭ་ཡ། མ་ཧཱ་བ་ཡ་བིརྻ་པ་ར་ཀྲ་མ། ཀོ་ཊ་ར་ཙ་ཡ། པྲ་མ་བྷ་བུ་བུ་མ། པྲ་མ་ས་རེ། བྷོ་བྷོ་ནཱ་ག་དི་པ་དི་པ་བྷི་ས་ར་མ་དྷ། པྲ་མ་དྷ་ན། ན་མུར་དྷ་ནན། སྤོ་ཊ་ཡ་ན། ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ། བྷ་ར་བྷ་ར། བྷརྣ་ཙཀྲ་དཱ་དཱ་ན། བྷོ་བ་ན། གེ་ན་དྲེ། མ་ཏི་ཀ་མ་དེ། བྷ་སྣ་ཙ་ཀྲེ་ན། ཤི་ན་ར་ཅི་ན་ད་མེ། ཧུ་ཏི་ཧུ་ཏི། ལི་ཧཱུཾ་ཧཱུྃ་ཕཊ། མུ་དུ་མུ་དུ། ཧ་ན་དུ་སརྦ་དུཥྚན། བྷ་སྦོ་དཱ་ཡ། ཧྲི་ད་ཡ། མ་ཧ་ཤྭ་རེ། ཛི་ལ་དི་ཏྲི་ཤུ་ད་ར་ད་ར། ཧ་ན་ཧ་ན། ཧཱུྃ་ཧཱུྃ་ཕཊ། མུ་ཙི་མུ་ཙི། དུ་ཙི་བ་བྷ་བ་བྷ་བ། ཀྵ་ནེ་ཀྵ་ནེ། ཙ་ཏུ་ར་ནཱན། མ་ཧཱ་རཱ་ཛཱ་ནན། ཛ་ལི་ཏི་ཤ་ག་ཏི། དྷ་རུ་དྷ་ར། ཕུ་ཕཊ། སྥུ་ར་སྥུ་ར། བྷྱོ་སྥུ་ར་བྷྱོ་སྥུ་ར། ཙ་ར་རྦ་ཙ་ར་རྦ། ཙ་ར་བྷ། ན་ཞུ་ཤུ་ཡ་དུ་སྟ་ནཱ་ག་ཧྲི་ད་ཡཱ། སྥུ་ཏ་ཡཱ། ཧཱུཾ་ཧཱུཾ་ཕཊ། དུ་སྟ་ཨ་ཤ་ནི། བྷད་ཏ་མ་ག་ཤི་བྱི་ལ་རྒྱ་ཙ་ལ། བིརྨ་ཀུ་རུ་ཧཱུཾ་ཧཱུྃ་ཕཊ། ཕཊ་སྭཱ་ཧཱ། ས་མ་ཡ་མ་ནུ་སྨ་ར། བྷོ་ནཱ་གི་ཏི་པ་ཏི་བྷྱོ་ཙན་དྷུ་ཡ་བྷོ་བྷོ་ཡ་བྷོ་བྷོ་དུ། ཏི་དུ་ཏི་དུ། ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ། བྷ་ཛི་ཡ་ཨ་ཤི་ནི། ན་ཤྱ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། བྷ་ཛྭ་ཡ་ཨ་ཤི་ན། ན་ཤྱ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། སརྦ་ནཱ་ག་ཧྲི་ད་ཡ་ནི། བི་སྥོ་ཊ་ཡ་ཀཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། བྷུ་ཏཱ་བ་ལོ་ཀི་ཏེ་ཡ་སྭཱ་ཧཱ། བུ་མ་ཡ་སྭཱ་ཧཱ། བྱིས་བྷུ་ར་ཡ་སྭཱ་ཧཱ། མ་ཧཱ་ཤ་ར་ཛྭ་ལི། ཏི་ཤུ་ལུ་ཡེ་སྭཱ་ཧཱ། ཙ་ཏུ་དྷ་མ་རེ་ཛ་ས་མ་ད་ར་ཡེ་སྭཱ་ཧཱ། བུ་ཙ་ཀ་ཡ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་དི་རི་དི་རི་སྭཱ་ཧཱ། ཛྭ་བ་ལི་ཧི་བྷ་ག་ཏི་ཡེ་སྭཱ་ཧཱ། ཛུ་ཏི་ཛུ་ཏི་སྭཱ་ཧཱ། ཧཱ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། རྡོ་རྗེ་འབར་བ་གནམ་ལྕགས་མཆུ་ཞེས་བྱ་བའི་གཟུངས་འདི་ཐོས་མ་ཐག་ཏུ། ཀླུ་མ་རུངས་པ་རྣམས་རུལ་ནས། དྲི་མི་ཞིམ་པ་རྣམས་མནམས་པར་འགྱུར་རོ། །ལུས་སོ་སོར་ཞིག་ཅིང་ཁྲོ་བར་གྱུར་ནས། བཅོམ་ལྡན་འདས་ཀྱི་ཞབས་ལ་གཏུགས་ཏེ། འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་བདག་ཅག་ལྷ་དང་བཅས་པའི་འཇིག་རྟེན་དུ་ཐམས་ཅད་ཀྱིས་སྨད་པར་གྱུར།

བདག་ཅག་ནི་མ་རུངས་པར་གྱུར་ཏོ། །བཅོམ་ལྡན་འདས་སླན་ཆད་འཛམ་བུའི་གླིང་གི་སེམས་ཅན་ཐམས་ཅད་ལ་གནོད་པར་མི་བགྱིད། འགུམ་པར་མི་བགྱིད། ཆུང་ངུ་ན་མིག་གིས་བལྟ་བར་ཡང་མི་བགྱིད་ན། སྐབས་བཙལ་བར་བ་ལྟ་ཅི་སྨོས། རྒྱ་མཚོའི་ཀླུའི་རྒྱལ་པོ་གླང་པོ་ལྟ་བུ་དང༌། ཀླུའི་རྒྱལ་པོ་གདུགས་ལྟ་བུ་དང༌། ཀླུའི་རྒྱལ་པོ་མཐུ་ཆེ་བ་བྱེ་བ་ཁྲག་ཁྲིག་བརྒྱ་སྟོང་གིས། བཅོམ་ལྡན་འདས་ཀྱི་ཞབས་ལ་མགོ་བོས་གཏུགས་ཏེ་ཕྱག་འཚལ་ནས། ཀླུ་ཐམས་ཅད་ཀྱིས་དུས་ཕྱི་མའི་ཚེ། འཛམ་བུའི་གླིང་གི་སེམས་ཅན་ཐམས་ཅད་ལ་མེ་ཏོག་དང༌། འབྲས་བུ་དང༌། ལོ་ཏོག་དང༌། ལོ་མ་དང༌། ཡལ་ག་ཐམས་ཅད་བསྲུང་བར་བགྱི། རླུང་དང༌། སེར་བ་དང༌། ཚ་བ་ཐམས་ཅད་གཅོད་པར་བགྱིད། དུས་དུས་སུ་ཆར་གྱི་རྒྱུན་ཡང་དབབ་པར་བགྱིའོ། །རང་གི་དམ་ཚིག་དང༌། ཡི་དམ་དང༌། དེ་བཞིན་གཤེགས་པའི་དམ་ཚིག་རྗེས་སུ་བསྲུང་བར་བགྱིའོ། །སེམས་ཅན་ཐམས་ཅད་ཀྱི་རེ་བ་ཡོངས་སུ་རྫོགས་པ་དང༌། ཡིད་ལ་བསམས་པ་ཡོངས་སུ་རྫོགས་པར་བགྱིད་པ་དང༌། གནོད་པར་མི་བགྱིད་དོ། །སྡོམ་པ་ཡང་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་ཀྱི་ཁྱད་པར་གྱི་ཡོན་ཏན་འདི་ལས་འགལ་བར་བགྱིད་ན། བདག་ཅག་ཀྱང་བྱང་ཆུབ་མཆོག་ཏུ་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་པར་མི་གྱུར། དེའི་བྲོ་ཚིག་ཀྱང་འདི་ལགས་ཏེ། ཏདྱ་ཐཱ། ཨོཾ་ཏ་ཀི་ཀི་ཏཿ གྷུ་ལི་གྷུ་ལི། མུ་ལི། ཤ་ཤ་གུ་གུ་ཤི་ཏི། བཱི་རུ་ཏི། ཧུ་ཡུ་ཧུ་ཡུ། ཧུ་རི་ཏི་ཧ། ཆར་ཆེ་བའམ། ཆར་མི་འབབ་པའམ། དུས་མ་ཡིན་པར་ཆར་བབས་པ་འམ། རླུང་དང་སྤྲིན་དང༌། དུས་མ་ཡིན་པའི་རླུང་དང༌། གནམ་ལྕགས་ལྟུང་བའི་ཚེ། རིག་འཛིན་གྱི་བྲོ་ཚིག་འདི་ཁང་ཐོག་གམ། ས་མཐོན་པོའི་སྟེང་ངམ། རིའི་སྟེང་ཡང་རུང༌། ཁ་ཤར་ཕྱོགས་སུ་བལྟས་ཏེ། ལན་བདུན་དུ་སྐད་ཆེན་པོར་བཟླས་ནས། ཡུངས་ཀར་ལ་ལན་བདུན་བཟླས་ཏེ་གྱེན་དུ་གཏོར་ན། གཏོར་མ་ཐག་ཏུ་ཀླུ་ཐམས་ཅད་དུམ་བུར་འགྱུར་ཞིང༌། ནན་ཏན་དུ་བྱས་པས་མྱུར་དུ་ཆར་འབེབས་སོ། །དྲིལ་བུ་ལ་ལན་ཉི་ཤུ་རྩ་གཅིག་བཟླས་ཏེ།

ཕྱོགས་བཞིར་ཁ་བལྟས་ནས་དཀྲོལ་ན། མཚམས་གཅོད་པར་འགྱུར་རོ། །རྡོ་རྗེ་ལ་ལན་བརྒྱ་རྩ་བརྒྱད་བཟླས་པས་རླུང་འགག་པར་བྱེད་དོ། །ལན་གཅིག་དྲན་པ་ཙམ་གྱིས་རླུང་དང༌། སེར་བ་དང༌། སྤྲིན་ཟློག་པར་བྱེད་པའོ། །གནམ་ལྕགས་ཀྱང་ཆུར་འགྱུར་རོ། །རྡོ་རྗེ་ཀླད་ལ་བསྐོར་ཞིང་ཕྱོགས་གང་དུ་བསྐོར་བ་དེར་སེར་བ་འབྱོལ་ལོ། །ནངས་པར་གནས་དབེན་པར་འདུག་སྟེ། ལན་ཉི་ཤུ་རྩ་གསུམ་དུ་བཟླས་ན་རླུང་དང༌། སེར་བ་དང་སྤྲིན་ཡང་བཀག་ཅིང་ཀླུས་གནོད་པར་མི་བྱེད་དེ། ཕྱིས་ཀྱང་མི་ཚུགས་སོ། །ཡུངས་ཀར་ལ་བསྔགས་ནས་ཕྱོགས་བཞིར་གཏོར་ན། ལོ་གཅིག་ཏུ་མཚམས་བཅད་དེ། རླུང་དང༌། སྤྲིན་ཀྱང་མི་གཡོས་ཏེ། གང་བསྒོ་བ་ཉན་ཅིང་རྒྱུན་དུ་བསྲུང་ཞིང་གསལ་བར་བྱེད་པར་འགྱུར་རོ། །གནོད་པ་བྱེད་པ་ཐམས་ཅད་བསལ་ཏོ། །ལེགས་པར་ཁྲུས་བྱས་ཏེ། སྐད་གསལ་བས་བཟླས་ཏེ། ཕྱོགས་བཞི་དང༌། བླ་འོག་ཏུ་ཡུངས་ཀར་གྱིས་གཏོར་ན། སེར་བ་དང༌། རླུང་དང༌། ཀླུ་གནོད་པར་བྱེད་པ་བཅིངས་པར་འགྱུར་རོ། །ལྟས་ངན་པ་དང༌། མི་དགེ་བ་དང༌། བཀྲ་མི་ཤིས་པ་དང༌། གཟའ་དང༌། འབྱུང་པོ་དང༌། ཤ་ཟ་དང༌། གནོད་སྦྱིན་དང༌། སྲིན་པོ་ཐམས་ཅད་བཅིངས་པར་འགྱུར་རོ། །དཔག་ཚད་འཁོར་ཡུག་ཏུ་མཚམས་བཅད་པར་འགྱུར་རོ། །བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་ཅེས་བཀའ་བསྩལ་ནས། ཚངས་པ་དང༌། འཇུག་སྲེད་དང༌། དབང་ཕྱུག་ལ་སོགས་པ་ཡི་རངས་ཏེ། མངོན་པར་བསྟོད་དོ། །རྡོ་རྗེ་གནམ་ལྕགས་མཆུ་ཞེས་བྱ་བའི་གཟུངས་རྫོགས་སོ།། །།ཞུས་དག །དགེའོ།། །།
