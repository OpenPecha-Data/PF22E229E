༄༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྻ་བི་ཛ་ཡ་བ་ཨ་བ་ཏི་ན་ནཱ་མ་པྲ་ཏྱཾ་གི་རཱ། བོད་སྐད་དུ། འཕགས་པ་ཕྱིར་བཟློག་པ་རྣམ་པར་རྒྱལ་བ་ཅན་ཞེས་བྱ་བ། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་འགྲོ་བ་ཐམས་ཅད་བསྲུང་བ་ལ་མངོན་པར་བརྩོན་པ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །གྲུབ་པ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །རིག་སྔགས་འཆང་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །སངས་རྒྱས་ལ་ཕྱག་འཚལ་ལོ། །ཆོས་ལ་ཕྱག་འཚལ་ལོ། །དགེ་འདུན་ལ་ཕྱག་འཚལ་ལོ། །བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་སྙིང་རྗེ་ཆེན་པོ་དང་ལྡན་པ་འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་ལ་ཕྱག་འཚལ་ལོ། །དེ་རྣམས་ལ་ཕྱག་འཚལ་ནས། ཕྱིར་ཟློག་པ་རྣམ་པར་རྒྱལ་བ་ཅན་ཞེས་བྱ་བ་གཤེད་བྱེད་དང་བྱད་སེལ་བ། རོ་ལངས་དང་བཀྲ་མི་ཤིས་པ་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་བྱེད་པ། རྒྱལ་པོའི་འཇིགས་པ་དང༌། ཆོམ་རྐུན་གྱི་འཇིགས་པ་དང༌། མེའི་འཇིགས་པ་དང༌། ཆུའི་འཇིགས་པ་དང༌། མཁའ་འགྲོ་མའི་འཇིགས་པ་དང༌། ཡི་དྭགས་དང༌། ཤ་ཟ་དང༌། གྲུལ་བུམ་གྱི་འཇིགས་པ་དང༌། གྲིབ་གནོན་གྱི་འཇིགས་པ་དང༌། བརྗེད་བྱེད་ཀྱི་འཇིགས་པ་དང༌། སྲུལ་པོའི་འཇིགས་པ་དང༌། ལུས་སྲུལ་པོའི་འཇིགས་པ་དང༌། ལམ་ལོག་པ་དང༌། ལམ་དུ་ཞུགས་པའི་འཇིགས་པ་དང༌། ཉིན་མོ་དང༌། མཚན་མོའི་འགྲོ་བ་སོ་སོའི་འཇིགས་པ་རྨུགས་པར་བྱེད་པ། རེངས་པར་བྱེད་པ། རྨོངས་པར་བྱེད་པ། མདོག་བཟང་པོར་བྱེད་པ། སྤྲོ་བ་ཡང་དག་པར་འཕེལ་བར་བྱེད་པ། དགྲ་ཐམས་ཅད་ཟློག་པར་བྱེད་པ། གེགས་དང་ལོག་འདྲེན་ཐམས་ཅད་ཞི་བར་བྱེད་པ་སྡིག་པ་ཐམས་ཅད་ཡང་དག་པར་བྱང་བར་བྱེད་པ། ཁ་རྨུགས་པར་བྱེད་པ། ཁ་རེངས་པར་བྱེད་པ། བཞིན་མཛེས་པར་བྱེད་པ་བརྗོད་པར་བྱའོ། །བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་སྙིང་རྗེ་ཆེན་པོ་དང་ལྡན་པ་འཕགས་པ་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་ལ་ཕྱག་འཚལ་ལོ། །གསང་སྔགས་ཀྱི་གཞི་འདི་དག་གྲུབ་པར་གྱུར་ཅིག །ཏདྱ་ཐཱ། ཨོཾ་དྷ་ར་དྷ་ར། དྷ་ར་དྷ་ར། དྷ་ར་དྷ་ར་དྷ་ར། དྷི་རི་དྷི་རི།

དྷི་རི་དྷི་རི། དྷི་རི་དྷི་རི་དྷི་རི། དྷུ་རུ་དྷུ་རུ། དྷུ་རུ་དྷུ་རུ། དྷུ་རུ་དྷུ་རུ་དྷུ་རུ། བདག་འཇིགས་པ་དང༌། གནོད་པ་ཐམས་ཅད་དང༌། གཤེད་བྱེད་དང༌། བྱད་སྟེམས་ཐམས་ཅད་ལས་བསྲུང་དུ་གསོལ། བསྲུང་དུ་གསོལ། ཏདྱ་ཐཱ། ས་ར་ས་ར་ས་ར་ས་ར་ས་ར་ས་ར་ས་ར། སི་རི་སི་རི་སི་རི་སི་རི་སི་རི་སི་རི་སི་རི། སུ་རུ་སུ་རུ་སུ་རུ་སུ་རུ་སུ་རུ་སུ་རུ་སུ་རུ། མ་ར་མ་ར་མ་ར་མ་ར་མ་ར་མ་ར་མ་ར། མི་རི་མི་རི་མི་རི་མི་རི་མི་རི་མི་རི་མི་རི། མུ་རུ་མུ་རུ་མུ་རུ་མུ་རུ་མུ་རུ་མུ་རུ་མུ་རུ། བདག་ནད་ཐམས་ཅད་ལས་བསྲུང་དུ་གསོལ། བསྲུང་དུ་གསོལ། ཏདྱ་ཐཱ། ཧི་ལི་ཧི་ལི་ཧི་ལི་ཧི་ལི་ཧི་ལི་ཧི་ལི་ཧི་ལི་ཧི་ལི། ཧུ་ལུ་ཧུ་ལུ་ཧུ་ལུ་ཧུ་ལུ་ཧུ་ལུ་ཧུ་ལུ་ཧུ་ལུ་ཧུ་ལུ། ཀི་ལི་ཀི་ལི་ཀི་ལི་ཀི་ལི་ཀི་ལི་ཀི་ལི་ཀི་ལི་ཀི་ལི་ཀི་ལི། མི་ལི་མི་ལི་མི་ལི་མི་ལི་མི་ལི་མི་ལི་མི་ལི་མི་ལི། ཙི་ལི་ཙི་ལི་ཙི་ལི་ཙི་ལི་ཙི་ལི་ཙི་ལི་ཙི་ལི་ཙི་ལི་ཙི་ལི། ཙུ་ལུ་ཙུ་ལུ་ཙུ་ལུ་ཙུ་ལུ་ཙུ་ལུ་ཙུ་ལུ་ཙུ་ལུ་ཙུ་ལུ་ཙུ་ལུ། འབྱུང་པོ་ཐམས་ཅད་བཟློག་པར་མཛད་དུ་གསོལ། ཨོཾ་བི་ས་ར་བི་ས་ར། ཀཾ་བ་ཀཾ་བ། ན་ཤ་ན་ཤ། ན་ཤ་ཡ་ན་ཤ་ཡ། དགྲ་གང་དག་བདག་ལ་མི་ཕན་པར་འདོད་པ་དང༌། རྣམ་པར་སྡང་བ་དང༌། གནོད་པར་སེམས་པ་དང༌། སྔགས་བྱེད་པ་དང༌། སྦྱིན་སྲེག་བྱེད་པ་དང༌། བྱེད་དུ་འཇུག་པ་དང༌། བྱས་པ་ལ་རྗེས་སུ་ཡི་རང་བ་དེ་དག་ཐམས་ཅད་ཉིད་ལ་སླར་སོང་ཞིག་སྭཱ་ཧཱ། མི་མཛའ་བ་ཐམས་ཅད་ཀྱི་མིག་བཅིང་ངོ༌། །རྣ་བ་བཅིང་ངོ༌། །སྣ་བཅིང་ངོ༌། །ལྕེ་བཅིང་ངོ༌། །ལུས་བཅིང་ངོ༌། །ཡིད་བཅིང་ངོ༌། །ཡན་ལག་དང་ཉིང་ལག་ཐམས་ཅད་བཅིང་ངོ༌། སྥུ་ཊ་སྥུ་ཊ་སྥུ་ཊ། སྥོ་ཊ་ཡ་སྥོ་ཊ་ཡ་སྤོ་ཊ་ཡ། གཤེད་བྱེད་དང་བྱད་ཐམས་ཅད་རྣམ་པར་བསལ་དུ་གསོལ། །རྣམ་པར་གཞོམ་དུ་གསོལ། །རྣམ་པར་གཞིག་ཏུ་གསོལ། །སརྦ་ཏྲུ་མ། ཏུ་རུ་ཏུ་རུ་ཏུ་རུ་ཏུ་རུ་སྭཱ་ཧཱ། བདག་འཇིགས་པ་དང་གནོད་པ་ཐམས་ཅད་དང༌།

གཤེད་བྱེད་དང༌། བྱད་དང༌། རོ་ལངས་དང༌། བརྗེད་བྱེད་ཐམས་ཅད་ལས་བསྲུང་དུ་གསོལ། བསྲུང་དུ་གསོལ། སྭཱ་ཧཱ། བདག་གི་སྡིག་པ་ཐམས་ཅད་བསལ་དུ་གསོལ། བསལ་དུ་གསོལ། ཙ་ར་ཙ་ར་སྭཱ་ཧཱ། མ་ར་མ་ར་སྭཱ་ཧཱ། སི་རི་སི་རི་སྭཱ་ཧཱ། ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། དྷུ་རུ་དྷུ་རུ་སྭཱ་ཧཱ། མི་གང་དག་གནོད་པར་འདོད་པ་ཕན་པར་མི་འདོད་ཅིང་བདག་ལ་ལྟ་བ་ཐམས་ཅད་རྨུགས་པར་བགྱིའོ། །རེངས་པར་བགྱིའོ། །འཁྲུགས་པར་བགྱིའོ། །རྨོངས་པར་བགྱིའོ། །འབྱུང་པོ་ཐམས་ཅད་དང༌། གདོན་ཐམས་ཅད་དང༌། ནད་ཐམས་ཅད་འཁྲུལ་བར་མཛད་མ། རྨོངས་པར་མཛད་མ། རྨུགས་པར་མཛད་མ། རེངས་པར་མཛད་མ། འཇིགས་པར་མཛད་མ། རྣམ་པར་འཇིགས་པར་མཛད་མ། རྣམ་པར་འཐོར་བར་མཛད་མ། རྣམ་པར་འཇོམས་པར་མཛད་མ་སྭཱ་ཧཱ། ནཱི་ལེ་ནཱི་ལེ་ནཱི་ལེ་ཀེ་ཤེ་སྭཱ་ཧཱ། པཱི་ཏི་པཱི་ཏི་པཱི་ཏི་ཀེ་ཤེ་སྭཱ་ཧཱ། ལོ་ཧི་ཏེ་ལོ་ཧི་ཏེ་ཀེ་ཤེ་སྭཱ་ཧཱ། ཨ་བ་དཱ་ཏེ། ཨ་བ་དཱ་ཏེ། ཨ་པ་དྷ་ཏ་ཀེ་ཤེ་སྭཱ་ཧཱ། ཤྭེ་ཏེ་ཤྭེ་ཏེ་བསྟྲ་དྷ་ར་ཎ་ཡེ་སྭཱ་ཧཱ། །འབྱུང་པོ་ཐམས་ཅད་དབང་བསྒྱུར་མ། །བདག་གི་སྡིག་པ་ཐམས་ཅད་བསལ་དུ་གསོལ། སྭཱ་ཧཱ། འཕེལ་བར་མཛད་མ། བདག་གི་དཔལ་དང་སྐལ་པ་བཟང་པོ་ཐམས་ཅད་སྤེལ་དུ་གསོལ། འཕེལ་བར་མཛད་དུ་གསོལ། སྭཱ་ཧཱ། བདག་འཇིགས་པ་དང༌། གནོད་པ་དང༌། ནད་འགོ་བ་དང༌། འཁྲུག་པ་ཐམས་ཅད་དང༌། གཤེད་བྱེད་དང༌། བྱད་ཐམས་ཅད་དང༌། ནད་ཐམས་ཅད་དང༌། གདོན་ཐམས་ཅད་དང༌། སྡིག་པ་ཐམས་ཅད་དང༌། བཟའ་ཉེས་པ་དང༌། སྒོམ་ཉེས་དང༌། གནོད་སྒྲིབ་དང༌། གནོད་ལྟས་ཐམས་ཅད་བསྲུང་དུ་གསོལ། བསྲུང་དུ་གསོལ། སྭཱ་ཧཱ། ཕྱིར་བཟློག་རྣམ་པར་རྒྱལ་བ་ཅན། །རྒྱལ་བ་འབྱུང་བ་གསུངས་པ་འདི། །གང་གིས་མགུལ་དུ་འདི་བཏགས་པ། །དེ་ནི་རྒྱལ་འགྱུར་ཐེ་ཚོམ་མེད། །སྡིག་པ་རྨི་ལམ་བཀྲ་མི་ཤིས། །རྒྱལ་པོ་མེ་དང་ཆོམ་རྐུན་དགྲ། །དེ་དག་རྣམས་ཀྱིས་དེ་མི་ཚུགས། །གང་གིས་མགུལ་དུ་འདི་བཏགས་པ། །དེ་ནི་ཆུ་ཡིས་ཁྱེར་མི་གྱུར། །སྡིག་པ་ཡང་དག་བྱང་བར་འགྱུར། །དགེ་བ་རྣམས་ནི་རྟག་ཏུ་འཕེལ། །ན་ད་མེད་པ་ཡང་ཐོབ་པར་འགྱུར། །འཕགས་པ་ཕྱིར་བཟློག་པ་རྣམ་པར་རྒྱལ་བ་ཅན་ཞེས་བྱ་བ་རྫོགས་སོ།། །།བཀྲ་ཤིས།། །།
